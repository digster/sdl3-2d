Recipe 08 — Audio Basics

Make a sound. The SDL3 audio-stream model in the smallest example that still teaches you what a stream is.

What you'll learn

  • SDL3's audio-stream design: queue PCM bytes in, the device pulls them out, the format is converted for you.
  • How to open the default playback device and bind a stream to it.
  • How to synthesise a beep in code so you don't need a WAV file.
  • Where SDL_LoadWAV would fit in if you do have one.

Prerequisites

Recipe 03 — Handling Input (for the keypress that triggers the beep).

Walkthrough

The audio-stream model

SDL2's old audio API had you supply a callback; SDL3 replaces that with audio streams. Conceptually: you push raw PCM bytes into a stream with SDL_PutAudioStreamData, the audio device consumes them in the background, and SDL handles any sample-rate / format / channel conversion in between. Your code never blocks on audio.

Init

Add SDL_INIT_AUDIO to the init flag and open a stream for the default playback device: 

if (!SDL_Init(SDL_INIT_VIDEO | SDL_INIT_AUDIO)) { /* ... */ }

// The format we'll PRODUCE. SDL converts to whatever the device wants.
SDL_AudioSpec want{};
want.freq     = 48000;
want.format   = SDL_AUDIO_F32;       // 32-bit float, easiest to synthesise
want.channels = 1;                    // mono

st->stream = SDL_OpenAudioDeviceStream(SDL_AUDIO_DEVICE_DEFAULT_PLAYBACK,
                                       &want, /*callback=*/nullptr, /*userdata=*/nullptr);
if (!st->stream) { SDL_Log("audio: %s", SDL_GetError()); return SDL_APP_FAILURE; }

// Devices start paused; resume so playback actually happens.
SDL_ResumeAudioStreamDevice(st->stream);

Synthesising a 0.15-second beep

static void play_beep(SDL_AudioStream *s, float freq_hz = 660.0f, float secs = 0.15f)
{
    const int rate    = 48000;
    const int samples = (int)(rate * secs);
    std::vector<float> buf(samples);

    // Simple sine with a 4ms linear attack/release so it doesn't click.
    const int env = 4 * rate / 1000;        // 4ms in samples
    for (int i = 0; i < samples; ++i) {
        float t = (float)i / rate;
        float a = 1.0f;
        if (i < env)             a = (float)i / env;
        if (i > samples - env)   a = (float)(samples - i) / env;
        buf[i] = a * 0.25f * SDL_sinf(2.0f * SDL_PI_F * freq_hz * t);
    }
    SDL_PutAudioStreamData(s, buf.data(), (int)(buf.size() * sizeof(float)));
}

The function returns immediately; the device drains buf in the background. Call it again to queue more — they queue, they don't overlap (we'd need a mixer for that).

Loading a WAV (if you have one)

// One-shot loader. Returns ownership; free with SDL_free(audio_buf).
SDL_AudioSpec spec; Uint8 *buf = nullptr; Uint32 len = 0;
if (!SDL_LoadWAV("assets/blip.wav", &spec, &buf, &len)) {
    SDL_Log("LoadWAV: %s", SDL_GetError());
}
// Open a stream with the SAME spec so no conversion is needed; or different,
// SDL will convert. Then SDL_PutAudioStreamData(stream, buf, (int)len).
SDL_free(buf);
The device starts paused. Forgetting SDL_ResumeAudioStreamDevice after open is the most common "I queued bytes but heard nothing" bug.

Complete listing

Press Space to beep at 660 Hz; arrow up/down beeps at 880/440 Hz.

/*
 * 08-audio-basics.cpp — a triphonic beep box.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

#include <vector>

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    SDL_AudioStream *stream = nullptr;
    int blink = 0;             // flashes the screen briefly when a beep plays
};

static void play_beep(SDL_AudioStream *s, float freq_hz, float secs = 0.15f)
{
    const int rate    = 48000;
    const int samples = (int)(rate * secs);
    std::vector<float> buf(samples);
    const int env = 4 * rate / 1000;
    for (int i = 0; i < samples; ++i) {
        float t = (float)i / rate;
        float a = 1.0f;
        if (i < env)             a = (float)i / env;
        if (i > samples - env)   a = (float)(samples - i) / env;
        buf[i] = a * 0.25f * SDL_sinf(2.0f * SDL_PI_F * freq_hz * t);
    }
    SDL_PutAudioStreamData(s, buf.data(), (int)(buf.size() * sizeof(float)));
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Audio Basics", "1.0", "com.example.sdl3.audio");
    if (!SDL_Init(SDL_INIT_VIDEO | SDL_INIT_AUDIO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Audio Basics", 640, 400,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);

    SDL_AudioSpec want{};
    want.freq     = 48000;
    want.format   = SDL_AUDIO_F32;
    want.channels = 1;
    st->stream = SDL_OpenAudioDeviceStream(SDL_AUDIO_DEVICE_DEFAULT_PLAYBACK,
                                           &want, nullptr, nullptr);
    if (!st->stream) { SDL_Log("audio: %s", SDL_GetError()); delete st; return SDL_APP_FAILURE; }
    SDL_ResumeAudioStreamDevice(st->stream);

    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN) {
        if (e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
        if (e->key.scancode == SDL_SCANCODE_SPACE)  { play_beep(st->stream, 660); st->blink = 8; }
        if (e->key.scancode == SDL_SCANCODE_UP)     { play_beep(st->stream, 880); st->blink = 8; }
        if (e->key.scancode == SDL_SCANCODE_DOWN)   { play_beep(st->stream, 440); st->blink = 8; }
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    Uint8 bg = st->blink > 0 ? 90 : 18;
    if (st->blink > 0) --st->blink;
    gfx::clear(st->renderer, bg, bg, bg + 10);

    // Three keys, three "buttons".
    const char *labels[3] = { "DOWN 440Hz", "SPACE 660Hz", "UP 880Hz" };
    (void)labels;  // (we'll render text in recipe 09)
    gfx::set_color(st->renderer, 255, 200, 60, 255);
    gfx::fill_rect(st->renderer,  60, 160, 160, 80);
    gfx::set_color(st->renderer, 120, 200, 255, 255);
    gfx::fill_rect(st->renderer, 240, 160, 160, 80);
    gfx::set_color(st->renderer, 255, 120, 200, 255);
    gfx::fill_rect(st->renderer, 420, 160, 160, 80);

    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        if (st->stream) SDL_DestroyAudioStream(st->stream);
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Change want.channels to 2 and synthesise a different waveform in each ear (interleave: L,R,L,R…). Now the beep is stereo.
  2. Save a WAV with your favourite tool, drop it next to the binary, replace the synthesiser with SDL_LoadWAV + SDL_PutAudioStreamData.
  3. Spam the keys. Notice beeps queue rather than overlap — you'd need to open multiple streams (one per voice) for chords.
  4. Add SDL_PauseAudioStreamDevice on P to toggle pause.
