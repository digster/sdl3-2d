Recipe 05 — Loading Textures

From shapes to images. Load a BMP, upload it to the GPU as a texture, and draw it — with the source/destination rect distinction explained.

What you'll learn

  • What an SDL_Surface is vs an SDL_Texture, and why both exist.
  • How to load a BMP with SDL_LoadBMP and upload it with SDL_CreateTextureFromSurface.
  • The role of src (where in the image) and dst (where on screen) rects in SDL_RenderTexture.
  • Why you free the surface immediately after uploading.

Prerequisites

Recipe 04 — Delta Time & Movement.

Walkthrough

Surfaces and textures

An SDL_Surface is a CPU-side pixel buffer — plain RAM you can poke. An SDL_Texture is the GPU-side equivalent — memory the renderer can sample at hardware speed. You load images into surfaces, then upload to textures for fast drawing. After upload, the surface is no longer needed.

SDL3 ships SDL_LoadBMP for BMP files only — no PNG, no JPG. That sounds restrictive but it keeps the dependency story honest: nothing extra to install. Most starter projects either convert their sprites to BMP or pull in SDL_image later for PNG/JPG.

Loading at startup

// Inside SDL_AppInit, after creating the renderer.
SDL_Surface *surf = SDL_LoadBMP("assets/player.bmp");
if (!surf) {
    SDL_Log("LoadBMP failed: %s", SDL_GetError());
    return SDL_APP_FAILURE;
}

// Optional: declare a colour-key so a magenta background becomes transparent.
SDL_SetSurfaceColorKey(surf, true, SDL_MapSurfaceRGB(surf, 255, 0, 255));

st->player_tex = SDL_CreateTextureFromSurface(st->renderer, surf);
SDL_DestroySurface(surf);                    // surface no longer needed
if (!st->player_tex) {
    SDL_Log("CreateTextureFromSurface: %s", SDL_GetError());
    return SDL_APP_FAILURE;
}

// Remember the dimensions for later.
SDL_GetTextureSize(st->player_tex, &st->player_w, &st->player_h);

Drawing the texture

SDL_RenderTexture takes three rects: the source (which part of the texture to sample), the destination (where on screen, including scale), or NULL for either to mean "the whole thing": 

// Full image at (200, 150), original size.
SDL_FRect dst = { 200.0f, 150.0f, st->player_w, st->player_h };
SDL_RenderTexture(st->renderer, st->player_tex, /*src=*/nullptr, &dst);

// Same image, scaled 2x on screen.
SDL_FRect dst2 = { 400.0f, 150.0f, st->player_w * 2, st->player_h * 2 };
SDL_RenderTexture(st->renderer, st->player_tex, nullptr, &dst2);

// Just the top-left 32x32 region, drawn at 600,150.
SDL_FRect src  = { 0, 0, 32, 32 };
SDL_FRect dst3 = { 600, 150, 32, 32 };
SDL_RenderTexture(st->renderer, st->player_tex, &src, &dst3);

Clean up

In SDL_AppQuit, free the texture before destroying the renderer (textures belong to a specific renderer): 

SDL_DestroyTexture(st->player_tex);
SDL_DestroyRenderer(st->renderer);
SDL_DestroyWindow(st->window);
No BMP yet? The complete listing below renders a small procedurally-generated checker texture so you can run the recipe with zero asset files. Swap in SDL_LoadBMP("yourfile.bmp") when you have one.

Complete listing

This version builds a 64×64 checkerboard surface in code, then uploads it — no asset file needed.

/*
 * 05-loading-textures.cpp — a procedurally-built checker texture,
 * drawn in three places at three sizes.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    SDL_Texture  *tex      = nullptr;
    float tw = 0, th = 0;
};

/* Build a 64x64 checker pattern, upload it, return the texture. */
static SDL_Texture *make_checker(SDL_Renderer *r)
{
    const int W = 64, H = 64, CELL = 8;
    SDL_Surface *s = SDL_CreateSurface(W, H, SDL_PIXELFORMAT_RGBA32);
    if (!s) return nullptr;

    Uint32 *px = static_cast<Uint32 *>(s->pixels);
    const SDL_PixelFormatDetails *pfd = SDL_GetPixelFormatDetails(s->format);
    for (int y = 0; y < H; ++y) {
        for (int x = 0; x < W; ++x) {
            bool on = ((x / CELL) + (y / CELL)) % 2 == 0;
            Uint8 v = on ? 230 : 60;
            px[y * (s->pitch / 4) + x] = SDL_MapRGBA(pfd, nullptr, v, v, v, 255);
        }
    }
    SDL_Texture *t = SDL_CreateTextureFromSurface(r, s);
    SDL_DestroySurface(s);
    return t;
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Loading Textures", "1.0", "com.example.sdl3.tex");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Loading Textures", 800, 600,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);

    st->tex = make_checker(st->renderer);
    if (!st->tex) { SDL_Log("checker: %s", SDL_GetError()); delete st; return SDL_APP_FAILURE; }
    SDL_GetTextureSize(st->tex, &st->tw, &st->th);

    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    gfx::clear(st->renderer, 18, 18, 28);

    // 1) Whole texture at original size.
    SDL_FRect dst1 = { 80,  120, st->tw, st->th };
    SDL_RenderTexture(st->renderer, st->tex, nullptr, &dst1);

    // 2) Same texture, drawn 3x larger.
    SDL_FRect dst2 = { 220, 120, st->tw * 3, st->th * 3 };
    SDL_RenderTexture(st->renderer, st->tex, nullptr, &dst2);

    // 3) Just the top-left quarter, drawn at 2x.
    SDL_FRect src  = { 0, 0, st->tw / 2, st->th / 2 };
    SDL_FRect dst3 = { 520, 120, st->tw, st->th };
    SDL_RenderTexture(st->renderer, st->tex, &src, &dst3);

    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        if (st->tex) SDL_DestroyTexture(st->tex);
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Save a 64×64 BMP to assets/sprite.bmp (relative to where the binary runs) and replace make_checker with SDL_LoadBMP.
  2. Try SDL_SetTextureScaleMode(st->tex, SDL_SCALEMODE_LINEAR) — the upscaled checker now blurs instead of staying crisp. SDL_SCALEMODE_NEAREST (the default) is what you want for pixel art.
  3. Apply a tint with SDL_SetTextureColorMod(st->tex, 255, 80, 80) — the texture's grey checkers turn red.
  4. Make a non-rectangular sprite: add a magenta border to your BMP and call SDL_SetSurfaceColorKey on the surface before uploading.
