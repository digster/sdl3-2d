Recipe 04 — Delta Time & Movement

Move things in pixels per second, not pixels per frame. The bug recipe 03 ended with, fixed once and for all.

What you'll learn

  • What "delta time" is and why a fixed step-per-frame is wrong.
  • How to measure dt with SDL_GetTicksNS.
  • How to express speeds in pixels per second and apply them with position += velocity * dt.
  • A simple bouncing-ball update that's frame-rate independent.

Prerequisites

Recipe 03 — Handling Input.

Walkthrough

Recipe 03 moved the player 4 pixels per frame. On a 60 Hz display that's 240 px/s. On a 144 Hz display that's 576 px/s. Same game, two completely different speeds. We don't want that.

The fix is the standard pattern: measure how long the last frame took, and scale all motion by that duration. The result is "pixels per second" — a real-world speed that doesn't care about refresh rate.

Measuring dt

SDL gives us a high-precision nanosecond timer. Stash the previous timestamp on the AppState and subtract: 

struct AppState {
    /* ... */
    Uint64 prev_ns = 0;          // last SDL_AppIterate's timestamp
    float  ball_x = 100, ball_y = 100;
    float  ball_vx = 220, ball_vy = 180;     // PIXELS PER SECOND, not per frame
    float  ball_r = 24;
};

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    Uint64 now_ns = SDL_GetTicksNS();
    float  dt     = static_cast<float>(now_ns - st->prev_ns) / 1.0e9f;  // ns -> s
    st->prev_ns   = now_ns;

    // Movement: position += velocity * elapsed time.
    st->ball_x += st->ball_vx * dt;
    st->ball_y += st->ball_vy * dt;

    /* ... bounce off the walls ... */
}

Initialise prev_ns at the end of SDL_AppInit with st->prev_ns = SDL_GetTicksNS();, otherwise the very first frame's dt will be a huge number (program-startup elapsed time) and the ball will teleport.

Bouncing off walls

After moving, clamp the position back inside the window and flip the velocity. Doing both prevents a "ball that escaped" bug where motion carried it past the wall and the next frame's dt-scaled velocity wasn't enough to get it back in. 

int win_w = 0, win_h = 0;
SDL_GetWindowSize(st->window, &win_w, &win_h);

if (st->ball_x - st->ball_r < 0)               { st->ball_x = st->ball_r;                       st->ball_vx = -st->ball_vx; }
if (st->ball_x + st->ball_r > (float)win_w)    { st->ball_x = (float)win_w - st->ball_r;        st->ball_vx = -st->ball_vx; }
if (st->ball_y - st->ball_r < 0)               { st->ball_y = st->ball_r;                       st->ball_vy = -st->ball_vy; }
if (st->ball_y + st->ball_r > (float)win_h)    { st->ball_y = (float)win_h - st->ball_r;        st->ball_vy = -st->ball_vy; }
Cap your dt. If your app is paused (laptop lid closed, debugger break), dt on the next frame will be huge and the ball will fly off-screen. A common safety net is dt = std::min(dt, 0.05f); — anything past 50 ms gets clamped. Real engines use a fixed-step accumulator; for a starter, the clamp is enough.

Complete listing

/*
 * 04-delta-time.cpp — a ball that bounces in pixels-per-second,
 * not pixels-per-frame.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

#include <algorithm>     // std::min

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    Uint64 prev_ns = 0;

    float ball_x = 140, ball_y = 180;
    float ball_vx = 220, ball_vy = 240;       // px/sec
    float ball_r = 26;
};

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Delta Time", "1.0", "com.example.sdl3.dt");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Delta Time", 800, 600,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);
    st->prev_ns = SDL_GetTicksNS();           // baseline so first dt isn't huge
    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    Uint64 now_ns = SDL_GetTicksNS();
    float  dt     = static_cast<float>(now_ns - st->prev_ns) / 1.0e9f;
    dt            = std::min(dt, 0.05f);      // pause-proof clamp
    st->prev_ns   = now_ns;

    int win_w = 0, win_h = 0;
    SDL_GetWindowSize(st->window, &win_w, &win_h);

    // Integrate position.
    st->ball_x += st->ball_vx * dt;
    st->ball_y += st->ball_vy * dt;

    // Bounce off walls.
    if (st->ball_x - st->ball_r < 0)            { st->ball_x = st->ball_r;                 st->ball_vx = -st->ball_vx; }
    if (st->ball_x + st->ball_r > (float)win_w) { st->ball_x = (float)win_w - st->ball_r;  st->ball_vx = -st->ball_vx; }
    if (st->ball_y - st->ball_r < 0)            { st->ball_y = st->ball_r;                 st->ball_vy = -st->ball_vy; }
    if (st->ball_y + st->ball_r > (float)win_h) { st->ball_y = (float)win_h - st->ball_r;  st->ball_vy = -st->ball_vy; }

    gfx::clear(st->renderer, 18, 18, 28);
    gfx::set_color(st->renderer, 255, 255, 255, 255);
    gfx::fill_circle(st->renderer, st->ball_x, st->ball_y, st->ball_r);
    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Disable vsync (SDL_SetRenderVSync(st->renderer, 0)) and confirm the ball moves at the same visible speed — it's now uncapped, but motion stays correct.
  2. Add gravity: st->ball_vy += 800.0f * dt; before the position update. The ball now arcs.
  3. Remove the std::min(dt, 0.05f) clamp, hit a debugger breakpoint for a few seconds, continue, and watch the ball teleport. Put the clamp back.
  4. Add a second ball with different colour and starting velocity. Notice how easy it is once the state lives in AppState.
