Recipe 10 — Tilemap Rendering

A 2D grid + a tileset = a level. With a scrolling camera, viewport culling, and one SDL_RenderTexture per visible tile.

What you'll learn

  • Storing a level as a flat std::vector<int> indexing into a tileset.
  • Mapping (tile column, tile row) → screen position with a camera offset.
  • Viewport culling — only iterating tiles you can actually see.
  • Why two tiles per "kind" sometimes outperforms one big texture lookup.

Prerequisites

Recipe 05 — Loading Textures and Recipe 04 for camera motion.

Walkthrough

The data layout

A tilemap is a 2D grid of small integers. 0 typically means "empty", and other values index into a tileset texture (a sprite sheet from recipe 06). Keeping the grid as a flat std::vector — not a vector-of-vectors — is faster and lays out cache-friendly: 

struct Tilemap {
    int cols, rows;
    std::vector<int> tiles;          // size = cols * rows; tiles[y*cols + x]
};

From map to screen

Tile (tx, ty) at tile size TS, with the camera at (cam_x, cam_y): 

float screen_x = tx * TS - cam_x;
float screen_y = ty * TS - cam_y;

And which tile is under a screen pixel (sx, sy)? 

int tx = (int)((sx + cam_x) / TS);
int ty = (int)((sy + cam_y) / TS);

Viewport culling

Drawing all 10,000 tiles in a level when only 200 fit on screen is wasteful. Compute the visible tile rectangle once, then iterate only that: 

int tx0 = std::max(0,                (int)(cam_x / TS));
int ty0 = std::max(0,                (int)(cam_y / TS));
int tx1 = std::min(map.cols,         (int)((cam_x + win_w) / TS) + 1);
int ty1 = std::min(map.rows,         (int)((cam_y + win_h) / TS) + 1);

for (int ty = ty0; ty < ty1; ++ty)
    for (int tx = tx0; tx < tx1; ++tx) {
        int id = map.tiles[ty * map.cols + tx];
        if (id == 0) continue;                  // skip empty
        draw_tile(renderer, tileset, id, tx*TS - cam_x, ty*TS - cam_y);
    }
Why a flat vector? Cache locality. The tiles you draw are adjacent in memory, so the CPU prefetcher catches up. A std::vector<std::vector<int>> would scatter each row across the heap and double the indirection.

Complete listing

Arrow keys / WASD scroll the camera over a procedurally-generated 60×40 map. Tileset is built in code — grass, stone, water — so no asset files. 

/*
 * 10-tilemap-rendering.cpp — scrollable tilemap with a procedural tileset.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

#include <vector>
#include <algorithm>

constexpr int TS    = 32;            // tile size in pixels
constexpr int COLS  = 60;
constexpr int ROWS  = 40;

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    SDL_Texture  *tileset  = nullptr;
    std::vector<int> map;            // size = COLS * ROWS
    float cam_x = 0, cam_y = 0;
    Uint64 prev_ns = 0;
};

/* Build a 3-tile tileset (32x96): empty, grass, stone, water arranged vertically. */
static SDL_Texture *make_tileset(SDL_Renderer *r)
{
    SDL_Surface *s = SDL_CreateSurface(TS, TS * 4, SDL_PIXELFORMAT_RGBA32);
    if (!s) return nullptr;
    Uint32 *px = static_cast<Uint32 *>(s->pixels);
    const SDL_PixelFormatDetails *pfd = SDL_GetPixelFormatDetails(s->format);

    auto fill = [&](int tile, Uint8 r_, Uint8 g_, Uint8 b_) {
        for (int y = 0; y < TS; ++y)
            for (int x = 0; x < TS; ++x) {
                // Subtle 1-pixel border so adjacent tiles read as a grid.
                bool border = (x == 0) || (y == 0);
                Uint8 v = border ? 30 : 0;
                px[(tile * TS + y) * (s->pitch / 4) + x] =
                    SDL_MapRGBA(pfd, nullptr,
                                (Uint8)std::max(0, r_ - v),
                                (Uint8)std::max(0, g_ - v),
                                (Uint8)std::max(0, b_ - v),
                                255);
            }
    };
    // tile 0 stays transparent (we'll skip it in the loop anyway).
    fill(1,  90, 160,  80);     // grass
    fill(2, 130, 130, 140);     // stone
    fill(3,  60, 110, 200);     // water

    SDL_Texture *t = SDL_CreateTextureFromSurface(r, s);
    SDL_DestroySurface(s);
    SDL_SetTextureScaleMode(t, SDL_SCALEMODE_NEAREST);
    return t;
}

/* Cheap "world" generator: grass everywhere with a stone ring and a water lake. */
static void generate_map(std::vector<int> &m)
{
    m.assign(COLS * ROWS, 1);
    for (int y = 0; y < ROWS; ++y)
        for (int x = 0; x < COLS; ++x) {
            int dx = x - COLS / 2, dy = y - ROWS / 2;
            int r2 = dx*dx + dy*dy;
            if (r2 > 14*14 && r2 < 18*18) m[y * COLS + x] = 2;  // stone ring
            if (r2 < 5*5)                  m[y * COLS + x] = 3;  // water lake
        }
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Tilemap", "1.0", "com.example.sdl3.tile");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Tilemap", 800, 600,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);
    st->tileset = make_tileset(st->renderer);
    generate_map(st->map);
    st->prev_ns = SDL_GetTicksNS();
    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    Uint64 now_ns = SDL_GetTicksNS();
    float dt = std::min((now_ns - st->prev_ns) / 1.0e9f, 0.05f);
    st->prev_ns = now_ns;

    const bool *k = SDL_GetKeyboardState(nullptr);
    const float v = 360.0f;
    if (k[SDL_SCANCODE_A] || k[SDL_SCANCODE_LEFT])  st->cam_x -= v * dt;
    if (k[SDL_SCANCODE_D] || k[SDL_SCANCODE_RIGHT]) st->cam_x += v * dt;
    if (k[SDL_SCANCODE_W] || k[SDL_SCANCODE_UP])    st->cam_y -= v * dt;
    if (k[SDL_SCANCODE_S] || k[SDL_SCANCODE_DOWN])  st->cam_y += v * dt;

    // Clamp camera to map bounds.
    int win_w = 0, win_h = 0;
    SDL_GetWindowSize(st->window, &win_w, &win_h);
    float max_cx = std::max(0.0f, (float)(COLS * TS - win_w));
    float max_cy = std::max(0.0f, (float)(ROWS * TS - win_h));
    st->cam_x = std::clamp(st->cam_x, 0.0f, max_cx);
    st->cam_y = std::clamp(st->cam_y, 0.0f, max_cy);

    gfx::clear(st->renderer, 12, 12, 16);

    int tx0 = std::max(0,    (int)(st->cam_x / TS));
    int ty0 = std::max(0,    (int)(st->cam_y / TS));
    int tx1 = std::min(COLS, (int)((st->cam_x + win_w) / TS) + 1);
    int ty1 = std::min(ROWS, (int)((st->cam_y + win_h) / TS) + 1);

    for (int ty = ty0; ty < ty1; ++ty) {
        for (int tx = tx0; tx < tx1; ++tx) {
            int id = st->map[ty * COLS + tx];
            if (id == 0) continue;
            SDL_FRect src = { 0, (float)(id * TS), (float)TS, (float)TS };
            SDL_FRect dst = { tx * (float)TS - st->cam_x,
                              ty * (float)TS - st->cam_y,
                              (float)TS, (float)TS };
            SDL_RenderTexture(st->renderer, st->tileset, &src, &dst);
        }
    }

    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        if (st->tileset) SDL_DestroyTexture(st->tileset);
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Print the tile count drawn each frame and toggle viewport culling off — observe the difference at the bigger map sizes.
  2. Add a second tilemap layer (e.g. "decorations"): same grid size, drawn over the base with 0 meaning "no overlay."
  3. Make the camera follow a "player" rectangle that moves with WASD — keep the player centred unless near the map edge.
  4. Load a CSV-style map file at startup instead of generating one in code: each line is a comma-separated row of ints.
