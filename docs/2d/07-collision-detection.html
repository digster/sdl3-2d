Recipe 07 — Collision Detection

The three checks that cover 95% of 2D games: AABB vs AABB, point-in-rect, and circle-vs-circle.

What you'll learn

  • Axis-aligned bounding box (AABB) intersection — the workhorse of 2D collision.
  • Why the "obvious" check has a single trick (separating axis on each side).
  • SDL3's built-in helpers (SDL_HasRectIntersectionFloat, SDL_PointInRectFloat) and when to skip them.
  • Circle-vs-circle with the squared-distance trick (no sqrt).

Prerequisites

Recipe 04 for moving things, and any of recipes 02/05 for drawing them.

Walkthrough

AABB vs AABB

An axis-aligned bounding box is a rectangle whose sides are parallel to the screen. Two AABBs overlap if and only if they overlap on both axes: 

struct Rect { float x, y, w, h; };

bool aabb_overlap(const Rect &a, const Rect &b)
{
    // Separating-axis test, one axis at a time.
    if (a.x + a.w <= b.x) return false;   // A is fully left of B
    if (b.x + b.w <= a.x) return false;   // B is fully left of A
    if (a.y + a.h <= b.y) return false;   // A is fully above B
    if (b.y + b.h <= a.y) return false;   // B is fully above A
    return true;
}

SDL3 has this built in if you use SDL_FRect: SDL_HasRectIntersectionFloat(&a, &b). Use it freely — it's the same four comparisons. Hand-rolling matters only when your rect type isn't SDL_FRect (e.g. has a centre + half-size).

Point in rect

A staple for mouse clicks on UI buttons. SDL ships SDL_PointInRectFloat; the by-hand version: 

bool point_in_rect(float px, float py, const Rect &r)
{
    return px >= r.x && px < r.x + r.w
        && py >= r.y && py < r.y + r.h;
}

Circle vs circle

Two circles overlap if the distance between their centres is less than the sum of their radii. Skip the sqrt by comparing squared distances — same answer, faster: 

bool circles_overlap(float ax, float ay, float ar,
                     float bx, float by, float br)
{
    float dx = ax - bx, dy = ay - by;
    float rsum = ar + br;
    return (dx * dx + dy * dy) <= (rsum * rsum);
}
"Touching" vs "overlapping". Decide your edge case deliberately: < means "must overlap strictly", <= means "touching also counts." For most games it doesn't matter; for physics, it does. Pick one and be consistent.

Complete listing

A box you steer with WASD, two static obstacles, and a roaming circle. Anything you're colliding with this frame paints red. 

/*
 * 07-collision-detection.cpp — WASD a player rectangle around a scene;
 * obstacles turn red when you touch them. A drifting circle does the
 * same against a fixed circle.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

#include <algorithm>

struct Rect { float x, y, w, h; };

static bool aabb_overlap(const Rect &a, const Rect &b)
{
    if (a.x + a.w <= b.x) return false;
    if (b.x + b.w <= a.x) return false;
    if (a.y + a.h <= b.y) return false;
    if (b.y + b.h <= a.y) return false;
    return true;
}

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    Uint64 prev_ns = 0;

    Rect player   = { 100, 100, 40, 40 };
    Rect obs1     = { 320, 180, 100, 60 };
    Rect obs2     = { 500, 320,  80, 80 };

    // Circle pair
    float cax = 180, cay = 430, car = 26;        // drifts
    float cbx = 520, cby = 470, cbr = 36;        // static
    float cax_v = 90.0f;                          // px/sec
};

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Collision Detection", "1.0", "com.example.sdl3.collide");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Collision Detection", 800, 600,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);
    st->prev_ns = SDL_GetTicksNS();
    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    Uint64 now_ns = SDL_GetTicksNS();
    float dt = std::min((now_ns - st->prev_ns) / 1.0e9f, 0.05f);
    st->prev_ns = now_ns;

    // WASD moves the player.
    const bool *keys = SDL_GetKeyboardState(nullptr);
    const float v = 260.0f;
    if (keys[SDL_SCANCODE_A]) st->player.x -= v * dt;
    if (keys[SDL_SCANCODE_D]) st->player.x += v * dt;
    if (keys[SDL_SCANCODE_W]) st->player.y -= v * dt;
    if (keys[SDL_SCANCODE_S]) st->player.y += v * dt;

    // The drifting circle bounces between x=120 and x=380.
    st->cax += st->cax_v * dt;
    if (st->cax < 120 || st->cax > 380) st->cax_v = -st->cax_v;

    bool hit1 = aabb_overlap(st->player, st->obs1);
    bool hit2 = aabb_overlap(st->player, st->obs2);

    float dx = st->cax - st->cbx, dy = st->cay - st->cby;
    float rsum = st->car + st->cbr;
    bool circle_hit = (dx*dx + dy*dy) <= (rsum * rsum);

    gfx::clear(st->renderer, 18, 18, 28);

    // Obstacles: red if colliding, otherwise the usual amber.
    gfx::set_color(st->renderer, hit1 ? 255 : 255, hit1 ? 80  : 200, hit1 ? 80  : 60, 255);
    gfx::fill_rect(st->renderer, st->obs1.x, st->obs1.y, st->obs1.w, st->obs1.h);
    gfx::set_color(st->renderer, hit2 ? 255 : 60,  hit2 ? 80  : 200, hit2 ? 80  : 120, 255);
    gfx::fill_rect(st->renderer, st->obs2.x, st->obs2.y, st->obs2.w, st->obs2.h);

    // Player.
    gfx::set_color(st->renderer, 120, 200, 255, 255);
    gfx::fill_rect(st->renderer, st->player.x, st->player.y, st->player.w, st->player.h);

    // Circles.
    gfx::set_color(st->renderer, circle_hit ? 255 : 255, circle_hit ? 80 : 120, circle_hit ? 80 : 200, 255);
    gfx::fill_circle(st->renderer, st->cax, st->cay, st->car);
    gfx::set_color(st->renderer, circle_hit ? 255 :  60, circle_hit ? 80 : 200, circle_hit ? 80 : 200, 255);
    gfx::fill_circle(st->renderer, st->cbx, st->cby, st->cbr);

    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Replace aabb_overlap with SDL_HasRectIntersectionFloat — convert Rect to SDL_FRect first (same field order, so a reinterpret_cast works in practice; use a tiny conversion function in production).
  2. Add response: when the player overlaps an obstacle, push it back by the smaller of the X / Y penetrations. This is the simplest "solid wall" behaviour.
  3. Replace circle-vs-circle with circle-vs-rect: find the rect's closest point to the circle's centre, then compare its distance to the radius.
  4. Spawn 50 small drifting boxes and call aabb_overlap for every pair. Confirm the cost is O(N²) — the motivation for spatial partitioning later.
