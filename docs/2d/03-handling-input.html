Recipe 03 — Handling Input

React to keys, the mouse, and window resizes inside SDL_AppEvent — and learn the right place to store "is the key held right now".

What you'll learn

  • How key / mouse / window events flow into SDL_AppEvent.
  • The difference between edge-triggered input (events) and state-polled input (per-frame).
  • How to use SDL_SCANCODE_* vs SDL_KEYCODE_* and when each is appropriate.
  • Why returning SDL_APP_SUCCESS from SDL_AppEvent is the right way to quit.

Prerequisites

Walkthrough

SDL3 calls our SDL_AppEvent function once for every event the windowing system hands it. Each event is a tagged union: e->type tells you which kind, and one of the union's sub-structs (e->key, e->button, e->motion, e->window …) holds the data.

Two flavours of input

Most game input falls into two patterns, and the distinction matters:

  • Edge events — "the user just pressed Space", "they just clicked". Handle these in SDL_AppEvent; they fire once per press. Good for jumps, shooting, menu selections.
  • Held state — "is the right-arrow being held this frame?". Don't reconstruct this from events. SDL maintains the live keyboard state for you; read it in SDL_AppIterate. Good for movement, charging, accelerating.

Edge events: key down, mouse click, quit

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);

    switch (e->type) {
    case SDL_EVENT_QUIT:
        return SDL_APP_SUCCESS;                   // window close button

    case SDL_EVENT_KEY_DOWN:
        // Scancode = physical key location (good for WASD movement);
        // Key      = logical character under the current layout (good for shortcuts).
        if (e->key.scancode == SDL_SCANCODE_ESCAPE) {
            return SDL_APP_SUCCESS;
        }
        if (e->key.scancode == SDL_SCANCODE_SPACE) {
            st->flashes++;                        // bump a counter for the demo
        }
        break;

    case SDL_EVENT_MOUSE_BUTTON_DOWN:
        st->last_click_x = e->button.x;
        st->last_click_y = e->button.y;
        SDL_Log("click %d at (%.0f, %.0f)",
                e->button.button, e->button.x, e->button.y);
        break;

    case SDL_EVENT_MOUSE_MOTION:
        st->mouse_x = e->motion.x;
        st->mouse_y = e->motion.y;
        break;

    case SDL_EVENT_WINDOW_RESIZED:
        SDL_Log("resized to %dx%d", e->window.data1, e->window.data2);
        break;

    default: break;
    }
    return SDL_APP_CONTINUE;
}

Held state: polling the keyboard each frame

For movement you want "while held". Don't track key-down/key-up manually — SDL keeps a live array for you: 

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    // SDL hands back a pointer into its internal table; do not free it.
    const bool *keys = SDL_GetKeyboardState(nullptr);

    const float speed = 240.0f;          // pixels per second (we'll fix dt in recipe 04)
    if (keys[SDL_SCANCODE_LEFT]  || keys[SDL_SCANCODE_A]) st->player_x -= speed / 60.0f;
    if (keys[SDL_SCANCODE_RIGHT] || keys[SDL_SCANCODE_D]) st->player_x += speed / 60.0f;
    if (keys[SDL_SCANCODE_UP]    || keys[SDL_SCANCODE_W]) st->player_y -= speed / 60.0f;
    if (keys[SDL_SCANCODE_DOWN]  || keys[SDL_SCANCODE_S]) st->player_y += speed / 60.0f;

    /* ... draw ... */
}
Don't quit with exit(). Returning SDL_APP_SUCCESS from any callback is the SDL3 way — it tells the callback runtime to wind down cleanly, triggering SDL_AppQuit and freeing your state.

Complete listing

/*
 * 03-handling-input.cpp — a draggable square that follows the mouse,
 * WASD/arrow movement, Space to flash a counter, Esc to quit.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    float player_x = 400.0f, player_y = 300.0f;
    float mouse_x  = 0.0f,   mouse_y  = 0.0f;
    float last_click_x = -1.0f, last_click_y = -1.0f;
    int   flashes = 0;
};

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Handling Input", "1.0", "com.example.sdl3.input");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("init: %s", SDL_GetError()); return SDL_APP_FAILURE; }

    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Handling Input", 800, 600,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("window: %s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);
    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);

    switch (e->type) {
    case SDL_EVENT_QUIT:               return SDL_APP_SUCCESS;
    case SDL_EVENT_KEY_DOWN:
        if (e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
        if (e->key.scancode == SDL_SCANCODE_SPACE)  st->flashes++;
        break;
    case SDL_EVENT_MOUSE_BUTTON_DOWN:
        st->last_click_x = e->button.x;
        st->last_click_y = e->button.y;
        SDL_Log("click %d at (%.0f, %.0f)", e->button.button, e->button.x, e->button.y);
        break;
    case SDL_EVENT_MOUSE_MOTION:
        st->mouse_x = e->motion.x;
        st->mouse_y = e->motion.y;
        break;
    default: break;
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    const bool *keys = SDL_GetKeyboardState(nullptr);
    const float step = 4.0f;
    if (keys[SDL_SCANCODE_LEFT]  || keys[SDL_SCANCODE_A]) st->player_x -= step;
    if (keys[SDL_SCANCODE_RIGHT] || keys[SDL_SCANCODE_D]) st->player_x += step;
    if (keys[SDL_SCANCODE_UP]    || keys[SDL_SCANCODE_W]) st->player_y -= step;
    if (keys[SDL_SCANCODE_DOWN]  || keys[SDL_SCANCODE_S]) st->player_y += step;

    gfx::clear(st->renderer, 18, 18, 28);

    // The "player" — WASD / arrows move it.
    gfx::set_color(st->renderer, 120, 200, 255, 255);
    gfx::fill_rect(st->renderer, st->player_x - 20, st->player_y - 20, 40, 40);

    // The mouse cursor, drawn as a crosshair.
    gfx::set_color(st->renderer, 255, 240, 120, 255);
    gfx::line(st->renderer, st->mouse_x - 8, st->mouse_y,     st->mouse_x + 8, st->mouse_y);
    gfx::line(st->renderer, st->mouse_x,     st->mouse_y - 8, st->mouse_x,     st->mouse_y + 8);

    // Last click marker (if any).
    if (st->last_click_x >= 0) {
        gfx::set_color(st->renderer, 255, 110, 90, 255);
        gfx::circle(st->renderer, st->last_click_x, st->last_click_y, 12);
    }

    // Space-pressed counter as a row of dots.
    gfx::set_color(st->renderer, 60, 200, 120, 255);
    for (int i = 0; i < st->flashes && i < 40; ++i) {
        gfx::fill_circle(st->renderer, 20.0f + i * 14.0f, 20.0f, 5.0f);
    }

    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Hold Space and notice the counter doesn't race — key-repeat fires events, but only on OS-level repeat intervals. For "fire while held," use SDL_GetKeyboardState instead.
  2. Swap a scancode check for a key.key == SDLK_W check and try it with an AZERTY-layout simulator. Scancode = physical key, key = logical character.
  3. Add a right-click case: e->button.button == SDL_BUTTON_RIGHT resets the player back to centre.
  4. The movement uses a fixed step per frame. Run on a 144 Hz display and watch the player rocket. We'll fix that in recipe 04.
