Recipe 06 — Sprite Animation

One texture, many frames. Build a sprite sheet in code, then march through it on a timer to animate.

What you'll learn

  • The "sprite sheet" idea — one texture holding a grid of frames.
  • Picking a frame with a moving src rect.
  • Driving the animation by an accumulator (frame-rate independent, like the ball in recipe 04).
  • The difference between looping and ping-pong animation.

Prerequisites

Recipe 05 — Loading Textures and the delta-time pattern from recipe 04.

Walkthrough

What a sprite sheet looks like

A sprite sheet is a grid: N frames laid out side by side in one image. If each frame is 32×32 and there are 8 frames in a row, the sheet is 256×32. Frame i lives at (i * 32, 0) with size 32×32. Animating means drawing only that frame's region of the texture, then advancing i on a timer.

State

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    SDL_Texture  *sheet    = nullptr;

    int frame_w = 32, frame_h = 32;
    int num_frames = 8;

    float frame_dur = 0.10f;     // seconds per frame -> 10 fps animation
    float frame_t   = 0.0f;      // accumulator
    int   frame_i   = 0;

    Uint64 prev_ns = 0;
};

Advancing the frame

// Each call to SDL_AppIterate:
Uint64 now_ns = SDL_GetTicksNS();
float dt = (now_ns - st->prev_ns) / 1.0e9f;
st->prev_ns = now_ns;

st->frame_t += dt;
while (st->frame_t >= st->frame_dur) {
    st->frame_t  -= st->frame_dur;
    st->frame_i   = (st->frame_i + 1) % st->num_frames;   // loop
}

Using a while (not if) means if the previous frame took longer than one animation step (e.g. a hitch), we catch up rather than slowing the animation down. Frame-rate independent.

Drawing the current frame

SDL_FRect src = {
    static_cast<float>(st->frame_i * st->frame_w),
    0.0f,
    static_cast<float>(st->frame_w),
    static_cast<float>(st->frame_h)
};
SDL_FRect dst = { 380.0f, 280.0f, st->frame_w * 4.0f, st->frame_h * 4.0f };  // 4x zoom
SDL_RenderTexture(st->renderer, st->sheet, &src, &dst);
Ping-pong (0→7→0) is one tweak away: track a direction (+1 / -1) and flip it when you hit either end. Useful for breathing, idling, or pendulum animations.

Complete listing

Builds an 8-frame "pulsing dot" sheet in code so the recipe runs with no asset files.

/*
 * 06-sprite-animation.cpp — a procedurally-built 8-frame "pulsing dot"
 * sheet driven by a delta-time accumulator.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    SDL_Texture  *sheet    = nullptr;
    int   frame_w = 32, frame_h = 32, num_frames = 8;
    float frame_dur = 0.10f, frame_t = 0.0f;
    int   frame_i = 0;
    Uint64 prev_ns = 0;
};

/* Paint an 8-frame sheet: each frame is a filled disc whose radius grows from
 * 4 to 14 then shrinks back. Demonstrates frame_i indexing into a horizontal strip. */
static SDL_Texture *make_sheet(SDL_Renderer *r, int fw, int fh, int n)
{
    SDL_Surface *s = SDL_CreateSurface(fw * n, fh, SDL_PIXELFORMAT_RGBA32);
    if (!s) return nullptr;
    Uint32 *px = static_cast<Uint32 *>(s->pixels);
    const SDL_PixelFormatDetails *pfd = SDL_GetPixelFormatDetails(s->format);

    for (int i = 0; i < n; ++i) {
        float t = (float)i / (n - 1);                   // 0..1
        float radius = 4.0f + 10.0f * (1.0f - std::abs(0.5f - t) * 2.0f); // peaks at i=4
        int cx = i * fw + fw / 2, cy = fh / 2;
        for (int y = 0; y < fh; ++y) {
            for (int x = 0; x < fw; ++x) {
                int dx = (i * fw + x) - cx, dy = y - cy;
                bool inside = dx*dx + dy*dy <= (int)(radius * radius);
                Uint32 c = inside
                    ? SDL_MapRGBA(pfd, nullptr, 255, 200, 80, 255)
                    : SDL_MapRGBA(pfd, nullptr, 0, 0, 0, 0);
                px[y * (s->pitch / 4) + (i * fw + x)] = c;
            }
        }
    }
    SDL_Texture *t = SDL_CreateTextureFromSurface(r, s);
    SDL_DestroySurface(s);
    SDL_SetTextureScaleMode(t, SDL_SCALEMODE_NEAREST);  // keep pixels crisp at 4x zoom
    return t;
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Sprite Animation", "1.0", "com.example.sdl3.anim");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Sprite Animation", 800, 600,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);
    st->sheet = make_sheet(st->renderer, st->frame_w, st->frame_h, st->num_frames);
    if (!st->sheet) { delete st; return SDL_APP_FAILURE; }
    st->prev_ns = SDL_GetTicksNS();
    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);

    Uint64 now_ns = SDL_GetTicksNS();
    float dt = (now_ns - st->prev_ns) / 1.0e9f;
    st->prev_ns = now_ns;

    st->frame_t += dt;
    while (st->frame_t >= st->frame_dur) {
        st->frame_t -= st->frame_dur;
        st->frame_i  = (st->frame_i + 1) % st->num_frames;
    }

    gfx::clear(st->renderer, 18, 18, 28);

    SDL_FRect src = {
        (float)(st->frame_i * st->frame_w), 0.0f,
        (float)st->frame_w, (float)st->frame_h
    };
    SDL_FRect dst = { 380, 280, st->frame_w * 4.0f, st->frame_h * 4.0f };
    SDL_RenderTexture(st->renderer, st->sheet, &src, &dst);

    // For comparison, also draw the whole sheet at the top so you can SEE the strip.
    SDL_FRect strip = { 20, 20, (float)(st->frame_w * st->num_frames * 2),
                                (float)(st->frame_h * 2) };
    SDL_RenderTexture(st->renderer, st->sheet, nullptr, &strip);

    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        SDL_DestroyTexture(st->sheet);
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Halve frame_dur to 0.05f — animation runs twice as fast without changing anything else.
  2. Swap the loop logic for ping-pong: keep a dir = +1, flip it at the ends.
  3. Drive frame_i from input instead of time: only advance when the user holds Space.
  4. Use a 2D sheet (rows = direction, columns = frame). Compute src.x = col * fw, src.y = row * fh.
