Recipe 11 — Particle Systems

Hundreds of tiny dots, each with their own physics and lifetime. The spawn / update / cull rhythm every particle engine boils down to.

What you'll learn

  • The minimal Particle struct: position, velocity, life, colour.
  • The spawn → integrate → cull loop, and why "swap-and-pop" beats erase.
  • How to fade particles by mapping life remaining to alpha.
  • Why we draw each particle as a small filled rect rather than a circle (and when to switch).

Prerequisites

Recipe 04 for the dt model, and any of recipes 02/05 for drawing.

Walkthrough

The particle

struct Particle {
    float x, y;        // position
    float vx, vy;      // velocity (px/sec)
    float life;        // seconds remaining
    float life_max;    // original life — used for fade ratio
    Uint8 r, g, b;
};

Spawning

On every click (or any event you like), push N particles into the vector with a random spread: 

static void spawn_burst(std::vector<Particle> &ps, float x, float y, int n)
{
    for (int i = 0; i < n; ++i) {
        float angle = SDL_randf() * 2.0f * SDL_PI_F;
        float speed = 80.0f + SDL_randf() * 220.0f;
        ps.push_back(Particle{
            x, y,
            SDL_cosf(angle) * speed, SDL_sinf(angle) * speed,
            0.8f + SDL_randf() * 0.6f,                       // life
            0.0f,                                            // life_max set below
            (Uint8)(180 + SDL_rand(76)),
            (Uint8)(100 + SDL_rand(155)),
            (Uint8)(40  + SDL_rand(80))
        });
        ps.back().life_max = ps.back().life;
    }
}

Integrating and culling

Walk the vector. Move each particle, age it, fade it. When one dies, swap with the last element and pop_back: O(1) deletion, no shifting. The trade-off is the order of particles isn't preserved — for visual particles that never matters. 

for (size_t i = 0; i < ps.size(); /* no ++ */) {
    Particle &p = ps[i];

    // Integrate.
    p.vy += 600.0f * dt;          // gravity
    p.x  += p.vx * dt;
    p.y  += p.vy * dt;
    p.life -= dt;

    if (p.life <= 0.0f) {
        ps[i] = ps.back();        // swap-and-pop
        ps.pop_back();
        continue;                 // don't ++i — i now points at a fresh element
    }
    ++i;
}

Drawing with fade

for (const Particle &p : ps) {
    Uint8 a = (Uint8)(255.0f * (p.life / p.life_max));
    gfx::set_color(renderer, p.r, p.g, p.b, a);
    gfx::fill_rect(renderer, p.x - 2, p.y - 2, 4, 4);
}

Mapping life / life_max to alpha gives a natural fade-out. A small filled rect is the cheapest GPU primitive; for hundreds of thousands you'd batch with SDL_RenderGeometry, but for the ~2,000 you'll spawn here, plain fill_rects are easy and fast enough.

Swap-and-pop, not erase. std::vector::erase(it) shifts every following element down by one — O(N) per removal. With many short-lived particles per frame this becomes the loop's hot spot. Swap-and-pop is O(1) and what every particle engine uses.

Complete listing

/*
 * 11-particle-systems.cpp — click to burst, watch particles arc and fade.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>
#include "gfx.hpp"

#include <vector>
#include <algorithm>

struct Particle {
    float x, y, vx, vy;
    float life, life_max;
    Uint8 r, g, b;
};

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    Uint64 prev_ns = 0;
    std::vector<Particle> ps;
    float mx = 400, my = 300;
};

static void spawn_burst(std::vector<Particle> &ps, float x, float y, int n)
{
    for (int i = 0; i < n; ++i) {
        float angle = SDL_randf() * 2.0f * SDL_PI_F;
        float speed = 80.0f + SDL_randf() * 220.0f;
        Particle p{
            x, y,
            SDL_cosf(angle) * speed, SDL_sinf(angle) * speed,
            0.8f + SDL_randf() * 0.6f, 0.0f,
            (Uint8)(180 + SDL_rand(76)),
            (Uint8)(100 + SDL_rand(155)),
            (Uint8)(40  + SDL_rand(80))
        };
        p.life_max = p.life;
        ps.push_back(p);
    }
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Particles", "1.0", "com.example.sdl3.particles");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Particles", 800, 600,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);
    SDL_SetRenderDrawBlendMode(st->renderer, SDL_BLENDMODE_BLEND);
    st->prev_ns = SDL_GetTicksNS();
    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_MOUSE_MOTION) { st->mx = e->motion.x; st->my = e->motion.y; }
    if (e->type == SDL_EVENT_MOUSE_BUTTON_DOWN) {
        spawn_burst(st->ps, e->button.x, e->button.y, 200);
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    Uint64 now_ns = SDL_GetTicksNS();
    float dt = std::min((now_ns - st->prev_ns) / 1.0e9f, 0.05f);
    st->prev_ns = now_ns;

    // Continuous emitter at the cursor — adds a steady stream too.
    if (SDL_GetMouseState(nullptr, nullptr) & SDL_BUTTON_MASK(SDL_BUTTON_LEFT)) {
        spawn_burst(st->ps, st->mx, st->my, 4);
    }

    int win_w = 0, win_h = 0;
    SDL_GetWindowSize(st->window, &win_w, &win_h);

    for (size_t i = 0; i < st->ps.size();) {
        Particle &p = st->ps[i];
        p.vy += 600.0f * dt;
        p.x  += p.vx * dt;
        p.y  += p.vy * dt;
        p.life -= dt;

        if (p.life <= 0.0f || p.y > (float)win_h + 20) {
            st->ps[i] = st->ps.back();
            st->ps.pop_back();
        } else {
            ++i;
        }
    }

    gfx::clear(st->renderer, 14, 14, 22);
    for (const Particle &p : st->ps) {
        Uint8 a = (Uint8)(255.0f * (p.life / p.life_max));
        gfx::set_color(st->renderer, p.r, p.g, p.b, a);
        gfx::fill_rect(st->renderer, p.x - 2, p.y - 2, 4, 4);
    }
    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Add air drag: p.vx *= (1.0f - 1.5f * dt); p.vy *= ...;. The fountain becomes "snowfall".
  2. Replace gravity with a target-attractor: each particle accelerates toward the mouse position.
  3. Switch the drawing to SDL_RenderGeometry — one big triangle list of two-triangle quads. Cheaper than thousands of fill_rects for very large counts.
  4. Add additive blending: SDL_BLENDMODE_ADD on a separate texture and they glow.
