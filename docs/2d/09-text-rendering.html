Recipe 09 — Text Rendering

Draw text without pulling in SDL_ttf. A 5×7 bitmap font in code is all you need for HUDs, scores, debug labels.

What you'll learn

  • The bitmap-font idea: each character is a small grid of on/off pixels.
  • How to turn an ASCII letter into a rectangle of gfx::point calls.
  • Why this is "good enough" for HUDs and where SDL_ttf takes over.

Prerequisites

Recipe 02 — Drawing Primitives.

Walkthrough

SDL3 ships no text-drawing API; you either pull in SDL_ttf for real font rendering, or you ship a bitmap font with the program. A bitmap font is just an image (or a table in code) of glyphs at a fixed cell size — for HUDs and debug overlays, it's plenty.

Encoding a glyph

Each character is a 5×7 grid; we encode it as 7 rows, each row a 5-bit mask. Bit set = filled pixel. The capital letter A: 

// Read bottom-up if you like, but I find top-down easier.
// .XXX.        0b01110
// X...X        0b10001
// X...X        0b10001
// XXXXX        0b11111   <-- crossbar
// X...X        0b10001
// X...X        0b10001
// X...X        0b10001
static const Uint8 GLYPH_A[7] = { 0b01110, 0b10001, 0b10001, 0b11111,
                                  0b10001, 0b10001, 0b10001 };

For a real font you'd encode every printable ASCII character; for this recipe we'll ship just A–Z, 0–9, space, period, colon and dash so the listing stays scannable.

Drawing one glyph

static void draw_glyph(SDL_Renderer *r, const Uint8 rows[7],
                       float x, float y, float scale)
{
    for (int row = 0; row < 7; ++row) {
        for (int col = 0; col < 5; ++col) {
            if (rows[row] & (1 << (4 - col))) {     // MSB is left-most
                gfx::fill_rect(r, x + col * scale, y + row * scale, scale, scale);
            }
        }
    }
}

Drawing each "on" pixel as a small filled rect means the same code renders crisply at any integer scale — a built-in zoom.

A string

static void draw_text(SDL_Renderer *r, const char *s,
                      float x, float y, float scale)
{
    while (*s) {
        const Uint8 *g = glyph_for(*s);          // returns nullptr for unknown chars
        if (g) draw_glyph(r, g, x, y, scale);
        x += 6.0f * scale;                       // 5px glyph + 1px gap
        ++s;
    }
}
When to upgrade. Bitmap fonts max out around debug overlays / arcade-style HUDs. The moment you want kerning, variable size, or non-Latin scripts, switch to SDL_ttf: it loads a TTF, asks FreeType to rasterise glyphs into a surface, and you upload the result like recipe 05.

static const Uint8 *glyph_for(char c)
{
    if (c >= 'a' && c <= 'z') c = (char)(c - 'a' + 'A');
    switch (c) {
    case 'A': return G_A; case 'B': return G_B; case 'C': return G_C; case 'D': return G_D;
    case 'E': return G_E; case 'F': return G_F; case 'G': return G_G; case 'H': return G_H;
    case 'I': return G_I; case 'J': return G_J; case 'K': return G_K; case 'L': return G_L;
    case 'M': return G_M; case 'N': return G_N; case 'O': return G_O; case 'P': return G_P;
    case 'Q': return G_Q; case 'R': return G_R; case 'S': return G_S; case 'T': return G_T;
    case 'U': return G_U; case 'V': return G_V; case 'W': return G_W; case 'X': return G_X;
    case 'Y': return G_Y; case 'Z': return G_Z;
    case '0': return G_0; case '1': return G_1; case '2': return G_2; case '3': return G_3;
    case '4': return G_4; case '5': return G_5; case '6': return G_6; case '7': return G_7;
    case '8': return G_8; case '9': return G_9;
    case '.': return G_DOT; case ':': return G_COL; case '-': return G_DASH;
    case ' ': return GLYPH_BLANK;
    default:  return nullptr;
    }
}

struct AppState {
    SDL_Window   *window   = nullptr;
    SDL_Renderer *renderer = nullptr;
    int score = 0;
};

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Text Rendering", "1.0", "com.example.sdl3.text");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("%s", SDL_GetError()); return SDL_APP_FAILURE; }
    AppState *st = new AppState();
    if (!SDL_CreateWindowAndRenderer("Text Rendering", 800, 400,
                                     SDL_WINDOW_RESIZABLE,
                                     &st->window, &st->renderer)) {
        SDL_Log("%s", SDL_GetError()); delete st; return SDL_APP_FAILURE;
    }
    SDL_SetRenderVSync(st->renderer, 1);
    *appstate = st;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *appstate, SDL_Event *e)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN) {
        if (e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
        if (e->key.scancode == SDL_SCANCODE_SPACE)  ++st->score;
    }
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    gfx::clear(st->renderer, 18, 18, 28);

    gfx::set_color(st->renderer, 240, 240, 240, 255);
    draw_text(st->renderer, "SDL3 RECIPES",                40,  40, 4.0f);
    draw_text(st->renderer, "PRESS SPACE TO SCORE.",       40, 110, 2.0f);

    char buf[32];
    SDL_snprintf(buf, sizeof buf, "SCORE: %d", st->score);
    gfx::set_color(st->renderer, 120, 200, 255, 255);
    draw_text(st->renderer, buf,                          40, 180, 3.0f);

    SDL_RenderPresent(st->renderer);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (st) {
        SDL_DestroyRenderer(st->renderer);
        SDL_DestroyWindow(st->window);
        delete st;
    }
}

Try it yourself

  1. Add lowercase letters by defining their own glyphs (don't reuse the uppercase shapes).
  2. Turn the glyph table into a real sprite-sheet texture (recipe 05) — one upload, one SDL_RenderTexture per character. Much faster than thousands of fill_rect calls when you have lots of text.
  3. Add an alignment helper: draw_text_right_aligned(...) that computes total width first.
  4. Try SDL_ttf for comparison: install brew install sdl3_ttf, link it, and call TTF_RenderText_Blended to produce a real anti-aliased surface.
