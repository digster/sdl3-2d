Recipe 04 — Index Buffers & Quads

A quad is two triangles. Four unique corners, six indices — index buffers in one sentence.

What you'll learn

  • Why duplicate vertices are bad and indices are the cure.
  • How to allocate, upload, and bind an SDL_GPU index buffer.
  • The two index sizes (16-bit vs 32-bit) and which to pick.
  • SDL_DrawGPUIndexedPrimitives as the indexed analogue of recipe 03's draw call.

Prerequisites

Recipe 03 — Vertex Buffers.

Walkthrough

The problem with no indices

Drawing a quad as two triangles in TRIANGLELIST mode with a plain vertex buffer takes six vertices — two of them duplicates of two others. For a cube it's even worse: 8 unique corners but 36 vertices in the list. Indices let you keep a small unique-corner array and reference each one as many times as you like.

Quad with 4 vertices and 6 indices

struct Vertex { float x, y, z, r, g, b, a; };

// Four corners of a unit quad centred on origin.
//   0 ─── 1
//   │  /  │
//   │ /   │
//   3 ─── 2
static const Vertex kQuad[4] = {
    { -0.5f,  0.5f, 0, 1, 0, 0, 1 },     // 0 top-left    red
    {  0.5f,  0.5f, 0, 0, 1, 0, 1 },     // 1 top-right   green
    {  0.5f, -0.5f, 0, 0, 0, 1, 1 },     // 2 bot-right   blue
    { -0.5f, -0.5f, 0, 1, 1, 0, 1 },     // 3 bot-left    yellow
};
// Two triangles: (0,1,2) and (0,2,3). The shared edge 0-2 is implicit.
static const Uint16 kIndices[6] = { 0, 1, 2,  0, 2, 3 };

Index buffer creation & upload

Same pattern as the vertex buffer in recipe 03, just with the INDEX usage flag. We can reuse the transfer-buffer helper if we like; the listing below shows the steps inline. 

SDL_GPUBufferCreateInfo ibi{};
ibi.usage = SDL_GPU_BUFFERUSAGE_INDEX;
ibi.size  = sizeof(kIndices);
st->ibuf  = SDL_CreateGPUBuffer(st->device, &ibi);

// ...build a transfer buffer of size sizeof(kIndices), memcpy kIndices in,

Drawing with indices

SDL_GPUBufferBinding vbind{}; vbind.buffer = st->vbuf; vbind.offset = 0;
SDL_BindGPUVertexBuffers(pass, 0, &vbind, 1);

SDL_GPUBufferBinding ibind{}; ibind.buffer = st->ibuf; ibind.offset = 0;
SDL_BindGPUIndexBuffer(pass, &ibind, SDL_GPU_INDEXELEMENTSIZE_16BIT);

SDL_DrawGPUIndexedPrimitives(pass,
    /*num_indices=*/   6,
    /*num_instances=*/ 1,
    /*first_index=*/   0,
    /*vertex_offset=*/ 0,
    /*first_instance=*/0);

16-bit vs 32-bit indices

SDL_GPU_INDEXELEMENTSIZE_16BIT can address up to 65,535 vertices and uses half the memory. 32BIT goes to ~4.3 billion. For most meshes 16-bit is the right default — switch only when a single buffer needs more than 64K vertices.

Winding order matters. By default SDL_GPU draws both sides of every triangle (no culling). If you enable back-face culling later, ensure your indices walk counter-clockwise when viewed from outside, or you'll get an invisible mesh.

Complete listing

/*
 * 04-index-buffers-and-quads.cpp — a four-corner quad drawn as two
 * indexed triangles. Same pipeline as recipe 03 plus an index buffer.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>

#include <cstring>
#include <cstddef>

struct Vertex { float x, y, z, r, g, b, a; };
static const Vertex kQuad[4] = {
    { -0.5f,  0.5f, 0, 1, 0, 0, 1 },
    {  0.5f,  0.5f, 0, 0, 1, 0, 1 },
    {  0.5f, -0.5f, 0, 0, 0, 1, 1 },
    { -0.5f, -0.5f, 0, 1, 1, 0, 1 },
};
static const Uint16 kIndices[6] = { 0, 1, 2,  0, 2, 3 };

static const char *kVS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSIn  { float3 pos [[attribute(0)]]; float4 color [[attribute(1)]]; };
struct VSOut { float4 position [[position]]; float4 color; };
vertex VSOut vertexMain(VSIn in [[stage_in]]) {
    VSOut o; o.position = float4(in.pos, 1.0); o.color = in.color; return o;
}
)msl";
static const char *kFS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float4 color; };
fragment float4 fragmentMain(VSOut in [[stage_in]]) { return in.color; }
)msl";

struct AppState {
    SDL_Window              *window   = nullptr;
    SDL_GPUDevice           *device   = nullptr;
    SDL_GPUGraphicsPipeline *pipeline = nullptr;
    SDL_GPUBuffer           *vbuf     = nullptr;
    SDL_GPUBuffer           *ibuf     = nullptr;
};

static SDL_AppResult fail(const char *w){ SDL_Log("%s: %s", w, SDL_GetError()); return SDL_APP_FAILURE; }

static SDL_GPUShader *make_shader(SDL_GPUDevice *d, const char *src,
                                  SDL_GPUShaderStage stage, const char *entry)
{
    SDL_GPUShaderCreateInfo i{};
    i.code = (const Uint8 *)src; i.code_size = std::strlen(src);
    i.entrypoint = entry; i.format = SDL_GPU_SHADERFORMAT_MSL; i.stage = stage;
    return SDL_CreateGPUShader(d, &i);
}

static bool upload(SDL_GPUDevice *dev, SDL_GPUBuffer *dst,
                   const void *src, Uint32 size)
{
    SDL_GPUTransferBufferCreateInfo tbi{};
    tbi.usage = SDL_GPU_TRANSFERBUFFERUSAGE_UPLOAD; tbi.size = size;
    SDL_GPUTransferBuffer *tb = SDL_CreateGPUTransferBuffer(dev, &tbi);
    if (!tb) return false;
    void *m = SDL_MapGPUTransferBuffer(dev, tb, false);
    std::memcpy(m, src, size);
    SDL_UnmapGPUTransferBuffer(dev, tb);
    SDL_GPUCommandBuffer *cmd = SDL_AcquireGPUCommandBuffer(dev);
    SDL_GPUCopyPass *cp = SDL_BeginGPUCopyPass(cmd);
    SDL_GPUTransferBufferLocation s{}; s.transfer_buffer = tb;
    SDL_GPUBufferRegion r{}; r.buffer = dst; r.size = size;
    SDL_UploadToGPUBuffer(cp, &s, &r, false);
    SDL_EndGPUCopyPass(cp);
    SDL_SubmitGPUCommandBuffer(cmd);
    SDL_ReleaseGPUTransferBuffer(dev, tb);
    return true;
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Index Buffers", "1.0", "com.example.sdl3.idx");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("init");
    AppState *st = new AppState(); *appstate = st;
    st->device = SDL_CreateGPUDevice(SDL_GPU_SHADERFORMAT_MSL, true, nullptr);
    if (!st->device) return fail("dev");
    st->window = SDL_CreateWindow("Index Buffers & Quads", 800, 600, SDL_WINDOW_RESIZABLE);
    SDL_ClaimWindowForGPUDevice(st->device, st->window);

    SDL_GPUBufferCreateInfo vbi{}; vbi.usage = SDL_GPU_BUFFERUSAGE_VERTEX; vbi.size = sizeof(kQuad);
    st->vbuf = SDL_CreateGPUBuffer(st->device, &vbi);
    SDL_GPUBufferCreateInfo ibi{}; ibi.usage = SDL_GPU_BUFFERUSAGE_INDEX;  ibi.size = sizeof(kIndices);
    st->ibuf = SDL_CreateGPUBuffer(st->device, &ibi);
    if (!upload(st->device, st->vbuf, kQuad, sizeof(kQuad))) return fail("upload v");
    if (!upload(st->device, st->ibuf, kIndices, sizeof(kIndices))) return fail("upload i");

    SDL_GPUShader *vs = make_shader(st->device, kVS, SDL_GPU_SHADERSTAGE_VERTEX,   "vertexMain");
    SDL_GPUShader *fs = make_shader(st->device, kFS, SDL_GPU_SHADERSTAGE_FRAGMENT, "fragmentMain");

    SDL_GPUVertexBufferDescription vb{}; vb.slot = 0;
    vb.pitch = sizeof(Vertex); vb.input_rate = SDL_GPU_VERTEXINPUTRATE_VERTEX;
    SDL_GPUVertexAttribute attrs[2]{};
    attrs[0].location = 0; attrs[0].buffer_slot = 0;
    attrs[0].format = SDL_GPU_VERTEXELEMENTFORMAT_FLOAT3; attrs[0].offset = offsetof(Vertex, x);
    attrs[1].location = 1; attrs[1].buffer_slot = 0;
    attrs[1].format = SDL_GPU_VERTEXELEMENTFORMAT_FLOAT4; attrs[1].offset = offsetof(Vertex, r);

    SDL_GPUVertexInputState vis{};
    vis.vertex_buffer_descriptions = &vb; vis.num_vertex_buffers = 1;
    vis.vertex_attributes = attrs;        vis.num_vertex_attributes = 2;

    SDL_GPUColorTargetDescription ct{};
    ct.format = SDL_GetGPUSwapchainTextureFormat(st->device, st->window);

    SDL_GPUGraphicsPipelineCreateInfo pci{};
    pci.vertex_shader = vs; pci.fragment_shader = fs;
    pci.primitive_type = SDL_GPU_PRIMITIVETYPE_TRIANGLELIST;
    pci.vertex_input_state = vis;
    pci.target_info.color_target_descriptions = &ct;
    pci.target_info.num_color_targets = 1;
    st->pipeline = SDL_CreateGPUGraphicsPipeline(st->device, &pci);
    SDL_ReleaseGPUShader(st->device, vs);
    SDL_ReleaseGPUShader(st->device, fs);
    if (!st->pipeline) return fail("pipeline");
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    SDL_GPUCommandBuffer *cmd = SDL_AcquireGPUCommandBuffer(st->device);
    SDL_GPUTexture *swap = nullptr; Uint32 sw = 0, sh = 0;
    SDL_WaitAndAcquireGPUSwapchainTexture(cmd, st->window, &swap, &sw, &sh);
    if (swap) {
        SDL_GPUColorTargetInfo cti{};
        cti.texture = swap;
        cti.clear_color = SDL_FColor{ 0.07f, 0.07f, 0.10f, 1 };
        cti.load_op = SDL_GPU_LOADOP_CLEAR; cti.store_op = SDL_GPU_STOREOP_STORE;
        SDL_GPURenderPass *pass = SDL_BeginGPURenderPass(cmd, &cti, 1, nullptr);
        SDL_BindGPUGraphicsPipeline(pass, st->pipeline);
        SDL_GPUBufferBinding vbind{}; vbind.buffer = st->vbuf;
        SDL_BindGPUVertexBuffers(pass, 0, &vbind, 1);
        SDL_GPUBufferBinding ibind{}; ibind.buffer = st->ibuf;
        SDL_BindGPUIndexBuffer(pass, &ibind, SDL_GPU_INDEXELEMENTSIZE_16BIT);
        SDL_DrawGPUIndexedPrimitives(pass, 6, 1, 0, 0, 0);
        SDL_EndGPURenderPass(pass);
    }
    SDL_SubmitGPUCommandBuffer(cmd);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (st->device) {
        if (st->ibuf) SDL_ReleaseGPUBuffer(st->device, st->ibuf);
        if (st->vbuf) SDL_ReleaseGPUBuffer(st->device, st->vbuf);
        SDL_ReleaseGPUGraphicsPipeline(st->device, st->pipeline);
        if (st->window) SDL_ReleaseWindowFromGPUDevice(st->device, st->window);
        SDL_DestroyGPUDevice(st->device);
    }
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Add 4 more vertices for a second smaller quad and append 6 indices that reference them. One draw call now renders both quads.
  2. Swap to SDL_GPU_INDEXELEMENTSIZE_32BIT and switch the array element type to Uint32. Same picture, double the index buffer size.
  3. Reorder the indices to {0, 2, 1, 0, 3, 2} — the winding flips. With back-face culling enabled later this would make the quad disappear.
