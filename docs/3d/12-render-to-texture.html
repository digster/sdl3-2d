Recipe 12 — Render-to-Texture

Capstone. Render the cube into an offscreen texture, then run a fullscreen post-process pass that samples it. The foundation of every screen-space effect.

What you'll learn

  • Allocating a colour-target texture (CPU-blind) for offscreen rendering.
  • Running two render passes in one frame: one to texture, one to screen.
  • The "fullscreen triangle" trick — three vertices, no buffer, covers the screen.
  • How a post-process shader transforms the rendered image (here, a vignette + tint).

Prerequisites

Recipe 08 — Textures & Samplers for the sampling side, and recipe 07 for the cube.

Walkthrough

Two passes, one frame

  1. Pass A — offscreen: the cube is rendered into scene_color (a regular SDL_GPUTexture with both COLOR_TARGET and SAMPLER usage flags).
  2. Pass B — screen: a single triangle covers the whole window, the fragment shader samples scene_color and applies a vignette + warm tint.

The scene target

SDL_GPUTextureCreateInfo ti{};
ti.type   = SDL_GPU_TEXTURETYPE_2D;
ti.format = SDL_GPU_TEXTUREFORMAT_R8G8B8A8_UNORM;
ti.usage  = SDL_GPU_TEXTUREUSAGE_COLOR_TARGET | SDL_GPU_TEXTUREUSAGE_SAMPLER;
ti.width  = w; ti.height = h;
ti.layer_count_or_depth = 1; ti.num_levels = 1;
ti.sample_count = SDL_GPU_SAMPLECOUNT_1;
st->scene_color = SDL_CreateGPUTexture(st->device, &ti);

Two usage flags is the key: COLOR_TARGET lets us draw into it, SAMPLER lets the next pass read it.

The fullscreen triangle

A common idiom: one huge triangle whose three vertices land beyond the viewport on two corners. The rasterizer clips it to the screen and you get exactly the right area covered with three vertices and zero buffers. UVs come for free if you compute them in the shader. 

// Fullscreen-triangle MSL vertex shader, indexed by vertex_id 0,1,2:
//   (-1,-1), (3,-1), (-1, 3)  → triangle that contains the [-1,1]² screen.
//   uv = position*0.5 + 0.5.
static const char *kPostVS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float2 uv; };
vertex VSOut postVS(uint vid [[vertex_id]]) {
    const float2 P[3] = { float2(-1,-1), float2(3,-1), float2(-1,3) };
    VSOut o;
    o.position = float4(P[vid], 0, 1);
    o.uv       = P[vid] * 0.5 + 0.5;
    o.uv.y     = 1.0 - o.uv.y;          // flip Y to match texture orientation
    return o;
}
)msl";

The post-process fragment shader

static const char *kPostFS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float2 uv; };
fragment float4 postFS(VSOut in [[stage_in]],
                       texture2d<float> scene [[texture(0)]],
                       sampler          smp   [[sampler(0)]]) {
    float4 col = scene.sample(smp, in.uv);
    // Warm tint.
    col.rgb *= float3(1.05, 0.95, 0.85);
    // Vignette: dim toward edges.
    float2 d = in.uv - 0.5;
    float vig = 1.0 - dot(d, d) * 1.6;
    col.rgb *= clamp(vig, 0.0, 1.0);
    return col;
}
)msl";

SDL_GPUColorTargetInfo b{};
b.texture     = swap;
b.clear_color = SDL_FColor{ 0, 0, 0, 1 };

SDL_BindGPUGraphicsPipeline(pb, st->post_pipeline);

SDL_BindGPUFragmentSamplers(pb, 0, &ts, 1);
SDL_DrawGPUPrimitives(pb, /*vertices=*/3, 1, 0, 0);
SDL_EndGPURenderPass(pb);

SDL_SubmitGPUCommandBuffer(cmd);
Why this matters. Render-to-texture is the foundation for: bloom, HDR tonemapping, depth-of-field, motion blur, deferred shading, SSAO, screen-space reflections — basically every effect with "screen" in the name. The plumbing is what you just wrote.

Complete listing

Spinning cube rendered offscreen, then a fullscreen-triangle pass applies a warm tint and vignette before the frame is presented. Listing is the longest in the track — almost everything is reused from earlier recipes, but bundled together for the first time. 

/*
 * 12-render-to-texture.cpp — two-pass frame: scene to offscreen texture,
 * then fullscreen post-process to the swapchain.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>

#include <cstring>
#include <cstddef>

struct Vertex { float x, y, z, r, g, b, a; };
static const Vertex kCube[24] = {
    {  0.5f,-0.5f,-0.5f, 1,0,0,1 },{  0.5f, 0.5f,-0.5f, 1,0,0,1 },{  0.5f, 0.5f, 0.5f, 1,0,0,1 },{  0.5f,-0.5f, 0.5f, 1,0,0,1 },
    { -0.5f,-0.5f, 0.5f, 0,1,1,1 },{ -0.5f, 0.5f, 0.5f, 0,1,1,1 },{ -0.5f, 0.5f,-0.5f, 0,1,1,1 },{ -0.5f,-0.5f,-0.5f, 0,1,1,1 },
    { -0.5f, 0.5f,-0.5f, 0,1,0,1 },{ -0.5f, 0.5f, 0.5f, 0,1,0,1 },{  0.5f, 0.5f, 0.5f, 0,1,0,1 },{  0.5f, 0.5f,-0.5f, 0,1,0,1 },
    { -0.5f,-0.5f, 0.5f, 1,0,1,1 },{ -0.5f,-0.5f,-0.5f, 1,0,1,1 },{  0.5f,-0.5f,-0.5f, 1,0,1,1 },{  0.5f,-0.5f, 0.5f, 1,0,1,1 },
    { -0.5f,-0.5f, 0.5f, 0,0,1,1 },{  0.5f,-0.5f, 0.5f, 0,0,1,1 },{  0.5f, 0.5f, 0.5f, 0,0,1,1 },{ -0.5f, 0.5f, 0.5f, 0,0,1,1 },
    {  0.5f,-0.5f,-0.5f, 1,1,0,1 },{ -0.5f,-0.5f,-0.5f, 1,1,0,1 },{ -0.5f, 0.5f,-0.5f, 1,1,0,1 },{  0.5f, 0.5f,-0.5f, 1,1,0,1 },
};
static const Uint16 kIdx[36] = {
    0,1,2, 4,5,6, 8,9,10, 12,13,14, 16,17,18, 20,21,22,
    0,2,3, 4,6,7, 8,10,11, 12,14,15, 16,18,19, 20,22,23,
};

struct Mat4 { float m[16]; };
static Mat4 mat4_identity(){ Mat4 r{}; r.m[0]=r.m[5]=r.m[10]=r.m[15]=1; return r; }
static Mat4 mat4_translate(float x,float y,float z){ Mat4 r=mat4_identity(); r.m[12]=x;r.m[13]=y;r.m[14]=z; return r;}
static Mat4 mat4_rotate_y(float a){ float c=SDL_cosf(a),s=SDL_sinf(a); Mat4 r=mat4_identity(); r.m[0]=c;r.m[2]=-s;r.m[8]=s;r.m[10]=c; return r;}
static Mat4 mat4_rotate_x(float a){ float c=SDL_cosf(a),s=SDL_sinf(a); Mat4 r=mat4_identity(); r.m[5]=c;r.m[6]=s;r.m[9]=-s;r.m[10]=c; return r;}
static Mat4 mat4_perspective(float fovy,float aspect,float n,float f){ float t=1/SDL_tanf(fovy*0.5f); Mat4 r{}; r.m[0]=t/aspect; r.m[5]=t; r.m[10]=f/(n-f); r.m[11]=-1; r.m[14]=(n*f)/(n-f); return r;}
static Mat4 mat4_mul(const Mat4 &a,const Mat4 &b){ Mat4 r{}; for(int c=0;c<4;++c) for(int row=0;row<4;++row){ float s=0; for(int k=0;k<4;++k) s+=a.m[k*4+row]*b.m[c*4+k]; r.m[c*4+row]=s; } return r;}

static const char *kSceneVS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSIn  { float3 pos [[attribute(0)]]; float4 color [[attribute(1)]]; };
struct VSOut { float4 position [[position]]; float4 color; };
struct UBO   { float4x4 mvp; };
vertex VSOut sceneVS(VSIn in [[stage_in]], constant UBO &u [[buffer(0)]]) {
    VSOut o; o.position = u.mvp * float4(in.pos, 1.0); o.color = in.color; return o;
}
)msl";
static const char *kSceneFS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float4 color; };
fragment float4 sceneFS(VSOut in [[stage_in]]) { return in.color; }
)msl";

static const char *kPostVS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float2 uv; };
vertex VSOut postVS(uint vid [[vertex_id]]) {
    const float2 P[3] = { float2(-1,-1), float2(3,-1), float2(-1,3) };
    VSOut o; o.position = float4(P[vid], 0, 1);
    o.uv = P[vid] * 0.5 + 0.5; o.uv.y = 1.0 - o.uv.y;
    return o;
}
)msl";
static const char *kPostFS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float2 uv; };
fragment float4 postFS(VSOut in [[stage_in]],
                       texture2d<float> scene [[texture(0)]],
                       sampler smp [[sampler(0)]]) {
    float4 col = scene.sample(smp, in.uv);
    col.rgb *= float3(1.05, 0.95, 0.85);
    float2 d = in.uv - 0.5;
    float vig = 1.0 - dot(d, d) * 1.6;
    col.rgb *= clamp(vig, 0.0, 1.0);
    return col;
}
)msl";

struct AppState {
    SDL_Window *window=nullptr; SDL_GPUDevice *device=nullptr;
    SDL_GPUGraphicsPipeline *scene_pipeline=nullptr;
    SDL_GPUGraphicsPipeline *post_pipeline=nullptr;
    SDL_GPUBuffer *vbuf=nullptr, *ibuf=nullptr;
    SDL_GPUTexture *scene_color=nullptr, *scene_depth=nullptr;
    SDL_GPUSampler *sampler=nullptr;
    Uint32 fb_w=0, fb_h=0;
    Uint64 start_ns=0;
};

static SDL_AppResult fail(const char *w){ SDL_Log("%s: %s", w, SDL_GetError()); return SDL_APP_FAILURE; }
static SDL_GPUShader *make_shader(SDL_GPUDevice *d, const char *src, SDL_GPUShaderStage s, const char *e,
                                  Uint32 ubo, Uint32 samplers){
    SDL_GPUShaderCreateInfo i{}; i.code=(const Uint8*)src; i.code_size=std::strlen(src);
    i.entrypoint=e; i.format=SDL_GPU_SHADERFORMAT_MSL; i.stage=s;
    i.num_uniform_buffers=ubo; i.num_samplers=samplers;
    return SDL_CreateGPUShader(d, &i);
}
static bool upload_buffer(SDL_GPUDevice *dev, SDL_GPUBuffer *dst, const void *src, Uint32 size){
    SDL_GPUTransferBufferCreateInfo tbi{}; tbi.usage=SDL_GPU_TRANSFERBUFFERUSAGE_UPLOAD; tbi.size=size;
    SDL_GPUTransferBuffer *tb=SDL_CreateGPUTransferBuffer(dev,&tbi);
    void *m=SDL_MapGPUTransferBuffer(dev,tb,false); std::memcpy(m,src,size); SDL_UnmapGPUTransferBuffer(dev,tb);
    SDL_GPUCommandBuffer *cmd=SDL_AcquireGPUCommandBuffer(dev);
    SDL_GPUCopyPass *cp=SDL_BeginGPUCopyPass(cmd);
    SDL_GPUTransferBufferLocation s{}; s.transfer_buffer=tb;
    SDL_GPUBufferRegion r{}; r.buffer=dst; r.size=size;
    SDL_UploadToGPUBuffer(cp,&s,&r,false);
    SDL_EndGPUCopyPass(cp); SDL_SubmitGPUCommandBuffer(cmd);
    SDL_ReleaseGPUTransferBuffer(dev,tb); return true;
}

static SDL_GPUTexture *create_color_target(SDL_GPUDevice *dev, Uint32 w, Uint32 h, SDL_GPUTextureFormat fmt){
    SDL_GPUTextureCreateInfo ti{}; ti.type=SDL_GPU_TEXTURETYPE_2D; ti.format=fmt;
    ti.usage = SDL_GPU_TEXTUREUSAGE_COLOR_TARGET | SDL_GPU_TEXTUREUSAGE_SAMPLER;
    ti.width=w; ti.height=h; ti.layer_count_or_depth=1; ti.num_levels=1; ti.sample_count=SDL_GPU_SAMPLECOUNT_1;
    return SDL_CreateGPUTexture(dev,&ti);
}
static SDL_GPUTexture *create_depth(SDL_GPUDevice *dev, Uint32 w, Uint32 h){
    SDL_GPUTextureCreateInfo ti{}; ti.type=SDL_GPU_TEXTURETYPE_2D;
    ti.format=SDL_GPU_TEXTUREFORMAT_D32_FLOAT; ti.usage=SDL_GPU_TEXTUREUSAGE_DEPTH_STENCIL_TARGET;
    ti.width=w; ti.height=h; ti.layer_count_or_depth=1; ti.num_levels=1; ti.sample_count=SDL_GPU_SAMPLECOUNT_1;
    return SDL_CreateGPUTexture(dev,&ti);
}

static void resize_offscreen(AppState *st, Uint32 w, Uint32 h)
{
    if (st->scene_color) SDL_ReleaseGPUTexture(st->device, st->scene_color);
    if (st->scene_depth) SDL_ReleaseGPUTexture(st->device, st->scene_depth);
    st->scene_color = create_color_target(st->device, w, h, SDL_GPU_TEXTUREFORMAT_R8G8B8A8_UNORM);
    st->scene_depth = create_depth      (st->device, w, h);
    st->fb_w = w; st->fb_h = h;
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Render to Texture", "1.0", "com.example.sdl3.rtt");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("init");
    AppState *st = new AppState(); *appstate = st;
    st->device = SDL_CreateGPUDevice(SDL_GPU_SHADERFORMAT_MSL, true, nullptr);
    st->window = SDL_CreateWindow("Render to Texture", 800, 600, SDL_WINDOW_RESIZABLE);
    SDL_ClaimWindowForGPUDevice(st->device, st->window);

    // --- geometry ---
    SDL_GPUBufferCreateInfo vbi{}; vbi.usage=SDL_GPU_BUFFERUSAGE_VERTEX; vbi.size=sizeof(kCube);
    st->vbuf=SDL_CreateGPUBuffer(st->device,&vbi);
    SDL_GPUBufferCreateInfo ibi{}; ibi.usage=SDL_GPU_BUFFERUSAGE_INDEX;  ibi.size=sizeof(kIdx);
    st->ibuf=SDL_CreateGPUBuffer(st->device,&ibi);
    upload_buffer(st->device, st->vbuf, kCube, sizeof(kCube));
    upload_buffer(st->device, st->ibuf, kIdx,  sizeof(kIdx));

    // --- sampler shared by post pass ---
    SDL_GPUSamplerCreateInfo si{}; si.min_filter=SDL_GPU_FILTER_LINEAR; si.mag_filter=SDL_GPU_FILTER_LINEAR;
    si.address_mode_u=SDL_GPU_SAMPLERADDRESSMODE_CLAMP_TO_EDGE;
    si.address_mode_v=SDL_GPU_SAMPLERADDRESSMODE_CLAMP_TO_EDGE;
    st->sampler = SDL_CreateGPUSampler(st->device, &si);

    // --- scene pipeline (offscreen) ---
    SDL_GPUShader *svs = make_shader(st->device, kSceneVS, SDL_GPU_SHADERSTAGE_VERTEX,   "sceneVS", 1, 0);
    SDL_GPUShader *sfs = make_shader(st->device, kSceneFS, SDL_GPU_SHADERSTAGE_FRAGMENT, "sceneFS", 0, 0);

    SDL_GPUVertexBufferDescription svb{}; svb.slot=0; svb.pitch=sizeof(Vertex); svb.input_rate=SDL_GPU_VERTEXINPUTRATE_VERTEX;
    SDL_GPUVertexAttribute sattrs[2]{};
    sattrs[0].location=0; sattrs[0].buffer_slot=0; sattrs[0].format=SDL_GPU_VERTEXELEMENTFORMAT_FLOAT3; sattrs[0].offset=offsetof(Vertex,x);
    sattrs[1].location=1; sattrs[1].buffer_slot=0; sattrs[1].format=SDL_GPU_VERTEXELEMENTFORMAT_FLOAT4; sattrs[1].offset=offsetof(Vertex,r);
    SDL_GPUVertexInputState svis{}; svis.vertex_buffer_descriptions=&svb; svis.num_vertex_buffers=1; svis.vertex_attributes=sattrs; svis.num_vertex_attributes=2;

    SDL_GPUColorTargetDescription sct{}; sct.format = SDL_GPU_TEXTUREFORMAT_R8G8B8A8_UNORM;
    SDL_GPUGraphicsPipelineCreateInfo spci{};
    spci.vertex_shader=svs; spci.fragment_shader=sfs;
    spci.primitive_type=SDL_GPU_PRIMITIVETYPE_TRIANGLELIST;
    spci.vertex_input_state=svis;
    spci.target_info.color_target_descriptions=&sct;
    spci.target_info.num_color_targets=1;
    spci.target_info.depth_stencil_format=SDL_GPU_TEXTUREFORMAT_D32_FLOAT;
    spci.target_info.has_depth_stencil_target=true;
    spci.depth_stencil_state.enable_depth_test=true;
    spci.depth_stencil_state.enable_depth_write=true;
    spci.depth_stencil_state.compare_op=SDL_GPU_COMPAREOP_LESS;
    st->scene_pipeline = SDL_CreateGPUGraphicsPipeline(st->device, &spci);
    SDL_ReleaseGPUShader(st->device, svs);
    SDL_ReleaseGPUShader(st->device, sfs);
    if (!st->scene_pipeline) return fail("scene pipeline");

    // --- post pipeline (to swapchain) ---
    SDL_GPUShader *pvs = make_shader(st->device, kPostVS, SDL_GPU_SHADERSTAGE_VERTEX,   "postVS", 0, 0);
    SDL_GPUShader *pfs = make_shader(st->device, kPostFS, SDL_GPU_SHADERSTAGE_FRAGMENT, "postFS", 0, 1);

    SDL_GPUColorTargetDescription pct{}; pct.format = SDL_GetGPUSwapchainTextureFormat(st->device, st->window);
    SDL_GPUGraphicsPipelineCreateInfo ppci{};
    ppci.vertex_shader=pvs; ppci.fragment_shader=pfs;
    ppci.primitive_type=SDL_GPU_PRIMITIVETYPE_TRIANGLELIST;
    // No vertex_input_state — we use [[vertex_id]] inside the shader.
    ppci.target_info.color_target_descriptions=&pct;
    ppci.target_info.num_color_targets=1;
    st->post_pipeline = SDL_CreateGPUGraphicsPipeline(st->device, &ppci);
    SDL_ReleaseGPUShader(st->device, pvs);
    SDL_ReleaseGPUShader(st->device, pfs);
    if (!st->post_pipeline) return fail("post pipeline");

    st->start_ns = SDL_GetTicksNS();
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    float t = (SDL_GetTicksNS() - st->start_ns) / 1.0e9f;

    SDL_GPUCommandBuffer *cmd = SDL_AcquireGPUCommandBuffer(st->device);
    SDL_GPUTexture *swap=nullptr; Uint32 sw=0, sh=0;
    SDL_WaitAndAcquireGPUSwapchainTexture(cmd, st->window, &swap, &sw, &sh);

    if (swap) {
        if (sw != st->fb_w || sh != st->fb_h) resize_offscreen(st, sw, sh);

        float aspect = sh > 0 ? (float)sw / (float)sh : 1.0f;
        Mat4 model = mat4_mul(mat4_rotate_y(t * 0.8f), mat4_rotate_x(t * 0.5f));
        Mat4 view  = mat4_translate(0, 0, -3.0f);
        Mat4 proj  = mat4_perspective(60.0f * SDL_PI_F / 180.0f, aspect, 0.1f, 100.0f);
        Mat4 mvp   = mat4_mul(proj, mat4_mul(view, model));

        /* Pass A: scene into scene_color */
        SDL_GPUColorTargetInfo a{};
        a.texture=st->scene_color; a.clear_color=SDL_FColor{0.05f,0.07f,0.10f,1};
        a.load_op=SDL_GPU_LOADOP_CLEAR; a.store_op=SDL_GPU_STOREOP_STORE;
        SDL_GPUDepthStencilTargetInfo da{};
        da.texture=st->scene_depth; da.clear_depth=1.0f;
        da.load_op=SDL_GPU_LOADOP_CLEAR; da.store_op=SDL_GPU_STOREOP_DONT_CARE;

        SDL_GPURenderPass *pa = SDL_BeginGPURenderPass(cmd, &a, 1, &da);
        SDL_BindGPUGraphicsPipeline(pa, st->scene_pipeline);
        SDL_GPUBufferBinding vbind{}; vbind.buffer=st->vbuf; SDL_BindGPUVertexBuffers(pa,0,&vbind,1);
        SDL_GPUBufferBinding ibind{}; ibind.buffer=st->ibuf; SDL_BindGPUIndexBuffer(pa,&ibind, SDL_GPU_INDEXELEMENTSIZE_16BIT);
        SDL_PushGPUVertexUniformData(cmd, 0, &mvp, sizeof(mvp));
        SDL_DrawGPUIndexedPrimitives(pa, 36, 1, 0, 0, 0);
        SDL_EndGPURenderPass(pa);

        /* Pass B: post-process into the swapchain */
        SDL_GPUColorTargetInfo b{};
        b.texture=swap; b.clear_color=SDL_FColor{0,0,0,1};
        b.load_op=SDL_GPU_LOADOP_CLEAR; b.store_op=SDL_GPU_STOREOP_STORE;
        SDL_GPURenderPass *pb = SDL_BeginGPURenderPass(cmd, &b, 1, nullptr);
        SDL_BindGPUGraphicsPipeline(pb, st->post_pipeline);
        SDL_GPUTextureSamplerBinding ts{}; ts.texture=st->scene_color; ts.sampler=st->sampler;
        SDL_BindGPUFragmentSamplers(pb, 0, &ts, 1);
        SDL_DrawGPUPrimitives(pb, 3, 1, 0, 0);
        SDL_EndGPURenderPass(pb);
    }

    SDL_SubmitGPUCommandBuffer(cmd);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (st->device) {
        if (st->sampler)     SDL_ReleaseGPUSampler(st->device, st->sampler);
        if (st->scene_color) SDL_ReleaseGPUTexture(st->device, st->scene_color);
        if (st->scene_depth) SDL_ReleaseGPUTexture(st->device, st->scene_depth);
        if (st->ibuf)        SDL_ReleaseGPUBuffer(st->device, st->ibuf);
        if (st->vbuf)        SDL_ReleaseGPUBuffer(st->device, st->vbuf);
        SDL_ReleaseGPUGraphicsPipeline(st->device, st->post_pipeline);
        SDL_ReleaseGPUGraphicsPipeline(st->device, st->scene_pipeline);
        if (st->window) SDL_ReleaseWindowFromGPUDevice(st->device, st->window);
        SDL_DestroyGPUDevice(st->device);
    }
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Switch the post shader to a black-and-white conversion: col.rgb = float3(dot(col.rgb, float3(0.299, 0.587, 0.114))).
  2. Make the offscreen target half the swapchain resolution. The post pass upscales it for free via the sampler — classic "render at low res, present at high res" trick.
  3. Add a third pass: a horizontal blur reading the scene texture, then a vertical blur reading the result. That's a separable Gaussian blur — the first step toward bloom.
  4. Combine with the FPS camera from recipe 11 for a fully post-processed walkthrough scene.

That's the 3D track. From here every screen-space effect — bloom, SSAO, DOF, motion blur — is a variation on the two-pass pattern in this recipe.
