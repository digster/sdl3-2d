Recipe 02 — Cross-Platform Shaders

MSL works on macOS; SPIR-V on Linux; DXIL on Windows. Detect what the device wants, ship whichever bytecode matches.

What you'll learn

  • The shader-format menu SDL_GPU understands: MSL, SPIR-V, DXIL, DXBC, MetalLib.
  • How to query the device for what it accepts (SDL_GetGPUShaderFormats).
  • The pragmatic recipe-track choice: MSL source for clarity on macOS, with a path to SPIR-V for portability.
  • Why offline shader compilation eventually becomes worth it.

Prerequisites

Recipe 01 — Hello Triangle.

Walkthrough

What "shader format" means

SDL_GPU does not contain a shader compiler. You hand it bytes that the backend's native graphics API can consume. The bit-flags are:

  • SDL_GPU_SHADERFORMAT_MSL — Metal Shading Language source text. macOS-only, compiled at runtime.
  • SDL_GPU_SHADERFORMAT_METALLIB — pre-compiled Metal bytecode. macOS, but faster startup.
  • SDL_GPU_SHADERFORMAT_SPIRV — Vulkan's IR. The lingua franca of cross-platform shaders.
  • SDL_GPU_SHADERFORMAT_DXIL — DirectX 12 intermediate language. Windows.
  • SDL_GPU_SHADERFORMAT_DXBC — older DirectX bytecode.

Negotiating with the device

When you create the device you advertise all the formats you can supply. SDL picks one and remembers it. You can then ask which format(s) the device ended up wanting: 

SDL_GPUDevice *dev = SDL_CreateGPUDevice(
    SDL_GPU_SHADERFORMAT_MSL | SDL_GPU_SHADERFORMAT_SPIRV,
    /*debug=*/true, /*name=*/nullptr);

SDL_GPUShaderFormat formats = SDL_GetGPUShaderFormats(dev);
if (formats & SDL_GPU_SHADERFORMAT_MSL)   SDL_Log("device wants MSL");
if (formats & SDL_GPU_SHADERFORMAT_SPIRV) SDL_Log("device wants SPIR-V");

Pattern: pick the right blob

For each shader, ship multiple bytecodes (compiled offline) plus a small dispatcher that picks the right one. With glslang/SPIRV-Cross you can author once in GLSL and emit SPIR-V, MSL, and DXIL: 

// Pseudo-API around what real engines do.
static SDL_GPUShader *load_shader(SDL_GPUDevice *dev,
                                  SDL_GPUShaderStage stage,
                                  const char *entrypoint)
{
    const Uint8 *bytes = nullptr;
    size_t       size  = 0;
    SDL_GPUShaderFormat fmt{};

    SDL_GPUShaderFormat have = SDL_GetGPUShaderFormats(dev);
    if (have & SDL_GPU_SHADERFORMAT_MSL)   { bytes = MSL_bytes;   size = MSL_size;   fmt = SDL_GPU_SHADERFORMAT_MSL;   }
    else if (have & SDL_GPU_SHADERFORMAT_SPIRV){ bytes = SPV_bytes;size = SPV_size;   fmt = SDL_GPU_SHADERFORMAT_SPIRV; }
    else if (have & SDL_GPU_SHADERFORMAT_DXIL) { bytes = DXIL_bytes;size = DXIL_size; fmt = SDL_GPU_SHADERFORMAT_DXIL;  }
    else { SDL_Log("no supported shader format"); return nullptr; }

    SDL_GPUShaderCreateInfo i{};
    i.code = bytes; i.code_size = size;
    i.entrypoint = entrypoint;
    i.format = fmt;
    i.stage  = stage;
    return SDL_CreateGPUShader(dev, &i);
}

Why this recipe track sticks to MSL source

Offline shader compilation would force us to ship a build pipeline: glslang, SPIRV-Cross, plus a way to embed the resulting blobs as C++ arrays. For a teaching template that pretends "just brew install sdl3" is enough, that's too big a leap. So:

  • On macOS we hand SDL MSL source and the Metal driver compiles at startup. Slight per-launch cost, zero extra tooling.
  • On Linux/Windows you'd add a SPIR-V/DXIL build step. The recipes below stay MSL because the repo targets macOS, but the dispatcher pattern above is what you'd add.
When to commit to offline. The moment you have more than two shaders, want symbol-level diagnostics, or distribute to non-developer machines, build a real shader-compile step. Authoring in GLSL → SPIRV-Cross → all three targets is the conventional path.

Complete listing

A diagnostic build that prints the available shader formats and then runs the same triangle as recipe 01.

/*
 * 02-cross-platform-shaders.cpp — diagnostic + MSL fallback.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>

#include <cstring>

static const char *kVS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float4 color; };
vertex VSOut vertexMain(uint vid [[vertex_id]]) {
    const float2 P[3] = { float2(0.0, 0.5), float2(-0.5,-0.5), float2(0.5,-0.5) };
    const float3 C[3] = { float3(1,0.4,0.3), float3(0.3,1,0.4), float3(0.4,0.3,1) };
    VSOut o; o.position = float4(P[vid], 0, 1); o.color = float4(C[vid], 1); return o;
}
)msl";

static const char *kFS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float4 color; };
fragment float4 fragmentMain(VSOut in [[stage_in]]) { return in.color; }
)msl";

struct AppState {
    SDL_Window              *window   = nullptr;
    SDL_GPUDevice           *device   = nullptr;
    SDL_GPUGraphicsPipeline *pipeline = nullptr;
};

static void log_formats(SDL_GPUShaderFormat f)
{
    SDL_Log("device accepts shader formats:");
    if (f & SDL_GPU_SHADERFORMAT_MSL)      SDL_Log("  - MSL (Metal source)");
    if (f & SDL_GPU_SHADERFORMAT_METALLIB) SDL_Log("  - METALLIB (compiled Metal)");
    if (f & SDL_GPU_SHADERFORMAT_SPIRV)    SDL_Log("  - SPIRV (Vulkan)");
    if (f & SDL_GPU_SHADERFORMAT_DXIL)     SDL_Log("  - DXIL (D3D12)");
    if (f & SDL_GPU_SHADERFORMAT_DXBC)     SDL_Log("  - DXBC (D3D11)");
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Cross-Platform Shaders", "1.0", "com.example.sdl3.shaders");
    if (!SDL_Init(SDL_INIT_VIDEO)) { SDL_Log("init: %s", SDL_GetError()); return SDL_APP_FAILURE; }

    AppState *st = new AppState(); *appstate = st;

    // Advertise EVERY format we could conceivably supply. SDL picks what's
    // appropriate for the actual GPU backend.
    SDL_GPUShaderFormat advertise =
        SDL_GPU_SHADERFORMAT_MSL | SDL_GPU_SHADERFORMAT_SPIRV | SDL_GPU_SHADERFORMAT_DXIL;

    st->device = SDL_CreateGPUDevice(advertise, true, nullptr);
    if (!st->device) { SDL_Log("CreateGPUDevice: %s", SDL_GetError()); return SDL_APP_FAILURE; }

    log_formats(SDL_GetGPUShaderFormats(st->device));

    st->window = SDL_CreateWindow("Cross-Platform Shaders", 800, 600, SDL_WINDOW_RESIZABLE);
    SDL_ClaimWindowForGPUDevice(st->device, st->window);

    // For this template we only carry MSL. Refuse if the device doesn't want it.
    if (!(SDL_GetGPUShaderFormats(st->device) & SDL_GPU_SHADERFORMAT_MSL)) {
        SDL_Log("This recipe only ships MSL shaders; non-Metal backend detected.");
        return SDL_APP_FAILURE;
    }

    SDL_GPUShaderCreateInfo vsi{};
    vsi.code = (const Uint8 *)kVS; vsi.code_size = std::strlen(kVS);
    vsi.entrypoint = "vertexMain"; vsi.format = SDL_GPU_SHADERFORMAT_MSL; vsi.stage = SDL_GPU_SHADERSTAGE_VERTEX;
    SDL_GPUShader *vs = SDL_CreateGPUShader(st->device, &vsi);

    SDL_GPUShaderCreateInfo fsi{};
    fsi.code = (const Uint8 *)kFS; fsi.code_size = std::strlen(kFS);
    fsi.entrypoint = "fragmentMain"; fsi.format = SDL_GPU_SHADERFORMAT_MSL; fsi.stage = SDL_GPU_SHADERSTAGE_FRAGMENT;
    SDL_GPUShader *fs = SDL_CreateGPUShader(st->device, &fsi);

    SDL_GPUColorTargetDescription ct{};
    ct.format = SDL_GetGPUSwapchainTextureFormat(st->device, st->window);

    SDL_GPUGraphicsPipelineCreateInfo pci{};
    pci.vertex_shader = vs; pci.fragment_shader = fs;
    pci.primitive_type = SDL_GPU_PRIMITIVETYPE_TRIANGLELIST;
    pci.target_info.color_target_descriptions = &ct;
    pci.target_info.num_color_targets = 1;
    st->pipeline = SDL_CreateGPUGraphicsPipeline(st->device, &pci);

    SDL_ReleaseGPUShader(st->device, vs);
    SDL_ReleaseGPUShader(st->device, fs);
    if (!st->pipeline) { SDL_Log("pipeline: %s", SDL_GetError()); return SDL_APP_FAILURE; }

    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    SDL_GPUCommandBuffer *cmd = SDL_AcquireGPUCommandBuffer(st->device);
    SDL_GPUTexture *swap = nullptr; Uint32 sw = 0, sh = 0;
    SDL_WaitAndAcquireGPUSwapchainTexture(cmd, st->window, &swap, &sw, &sh);
    if (swap) {
        SDL_GPUColorTargetInfo cti{};
        cti.texture = swap;
        cti.clear_color = SDL_FColor{ 0.05f, 0.07f, 0.10f, 1 };
        cti.load_op = SDL_GPU_LOADOP_CLEAR; cti.store_op = SDL_GPU_STOREOP_STORE;
        SDL_GPURenderPass *pass = SDL_BeginGPURenderPass(cmd, &cti, 1, nullptr);
        SDL_BindGPUGraphicsPipeline(pass, st->pipeline);
        SDL_DrawGPUPrimitives(pass, 3, 1, 0, 0);
        SDL_EndGPURenderPass(pass);
    }
    SDL_SubmitGPUCommandBuffer(cmd);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (st->device) {
        SDL_ReleaseGPUGraphicsPipeline(st->device, st->pipeline);
        if (st->window) SDL_ReleaseWindowFromGPUDevice(st->device, st->window);
        SDL_DestroyGPUDevice(st->device);
    }
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Run on macOS and check the console: you should see "MSL" and possibly "METALLIB."
  2. Remove SDL_GPU_SHADERFORMAT_MSL from the advertise bitmask. Device creation fails on macOS because no usable format was offered.
  3. Install glslangValidator and compile a small GLSL shader to SPIR-V. Embed the bytes as a static const Uint8 SPV_VS[] = { ... } array and add a SPIRV branch to your loader.
