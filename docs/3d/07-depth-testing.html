Recipe 07 — Depth Testing

Fix the broken cube. Allocate a depth texture, attach it to the render pass, and let the GPU sort faces for you.

What you'll learn

  • What the depth buffer is and why every 3D scene needs one.
  • How to allocate an SDL_GPUTexture with a depth format.
  • Two pipeline changes: depth-target format, and depth-stencil state.
  • Two render-pass changes: bind the depth attachment, clear it to 1.0.

Prerequisites

Recipe 06 — The 3D Cube.

Walkthrough

The idea

Alongside the colour render target, the GPU also keeps a depth target: one float per pixel that records how far away the closest drawn pixel is. When a new triangle's pixel arrives, the GPU compares its depth to whatever's already there. If the new one is further, it's discarded — the closer pixel wins. End of overdraw bug.

Create a depth texture

Allocate it at the current swapchain size. If the window resizes, you re-create — for this recipe we'll handle resize lazily by tracking the last-known size. 

static SDL_GPUTexture *create_depth(SDL_GPUDevice *dev, Uint32 w, Uint32 h)
{
    SDL_GPUTextureCreateInfo ti{};
    ti.type                = SDL_GPU_TEXTURETYPE_2D;
    ti.format              = SDL_GPU_TEXTUREFORMAT_D32_FLOAT;
    ti.usage               = SDL_GPU_TEXTUREUSAGE_DEPTH_STENCIL_TARGET;
    ti.width               = w;
    ti.height              = h;
    ti.layer_count_or_depth= 1;
    ti.num_levels          = 1;
    ti.sample_count        = SDL_GPU_SAMPLECOUNT_1;
    return SDL_CreateGPUTexture(dev, &ti);
}

Pipeline changes

Tell the pipeline which depth format you'll use, then enable depth-test + depth-write with the standard "less than" comparison: 

pci.target_info.depth_stencil_format = SDL_GPU_TEXTUREFORMAT_D32_FLOAT;
pci.target_info.has_depth_stencil_target = true;

pci.depth_stencil_state.enable_depth_test  = true;
pci.depth_stencil_state.enable_depth_write = true;
pci.depth_stencil_state.compare_op         = SDL_GPU_COMPAREOP_LESS;

Render-pass changes

SDL_GPUDepthStencilTargetInfo dti{};
dti.texture     = st->depth;
dti.clear_depth = 1.0f;                          // farthest possible
dti.load_op     = SDL_GPU_LOADOP_CLEAR;
dti.store_op    = SDL_GPU_STOREOP_DONT_CARE;     // we don't read it back

SDL_GPURenderPass *pass = SDL_BeginGPURenderPass(cmd, &cti, 1, &dti);

Note the depth-target's store_op is DONT_CARE — we don't need the depth values after the pass finishes (we're not going to sample them). That hint lets the driver elide a write to main memory.

Handling resize

We re-create the depth texture only when the swapchain reports a different size than the one we used last time: 

if (sw != st->depth_w || sh != st->depth_h) {
    SDL_ReleaseGPUTexture(st->device, st->depth);
    st->depth   = create_depth(st->device, sw, sh);
    st->depth_w = sw;
    st->depth_h = sh;
}
Z range conventions. SDL_GPU normalises depth to [0, 1], where 0 is the near plane and 1 is the far plane. Clear to 1.0 so that "no geometry yet" is the furthest possible value, and any real geometry wins the depth test.

Complete listing

The recipe-06 cube, plus everything above.

/*
 * 07-depth-testing.cpp — the 3D cube with a depth attachment that actually
 * sorts faces correctly.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>

#include <cstring>
#include <cstddef>

struct Vertex { float x, y, z, r, g, b, a; };
static const Vertex kCube[24] = {
    {  0.5f,-0.5f,-0.5f, 1,0,0,1 },{  0.5f, 0.5f,-0.5f, 1,0,0,1 },{  0.5f, 0.5f, 0.5f, 1,0,0,1 },{  0.5f,-0.5f, 0.5f, 1,0,0,1 },
    { -0.5f,-0.5f, 0.5f, 0,1,1,1 },{ -0.5f, 0.5f, 0.5f, 0,1,1,1 },{ -0.5f, 0.5f,-0.5f, 0,1,1,1 },{ -0.5f,-0.5f,-0.5f, 0,1,1,1 },
    { -0.5f, 0.5f,-0.5f, 0,1,0,1 },{ -0.5f, 0.5f, 0.5f, 0,1,0,1 },{  0.5f, 0.5f, 0.5f, 0,1,0,1 },{  0.5f, 0.5f,-0.5f, 0,1,0,1 },
    { -0.5f,-0.5f, 0.5f, 1,0,1,1 },{ -0.5f,-0.5f,-0.5f, 1,0,1,1 },{  0.5f,-0.5f,-0.5f, 1,0,1,1 },{  0.5f,-0.5f, 0.5f, 1,0,1,1 },
    { -0.5f,-0.5f, 0.5f, 0,0,1,1 },{  0.5f,-0.5f, 0.5f, 0,0,1,1 },{  0.5f, 0.5f, 0.5f, 0,0,1,1 },{ -0.5f, 0.5f, 0.5f, 0,0,1,1 },
    {  0.5f,-0.5f,-0.5f, 1,1,0,1 },{ -0.5f,-0.5f,-0.5f, 1,1,0,1 },{ -0.5f, 0.5f,-0.5f, 1,1,0,1 },{  0.5f, 0.5f,-0.5f, 1,1,0,1 },
};
static const Uint16 kIdx[36] = {
    0,1,2, 4,5,6, 8,9,10, 12,13,14, 16,17,18, 20,21,22,
    0,2,3, 4,6,7, 8,10,11, 12,14,15, 16,18,19, 20,22,23,
};

struct Mat4 { float m[16]; };
static Mat4 mat4_identity(){ Mat4 r{}; r.m[0]=r.m[5]=r.m[10]=r.m[15]=1; return r; }
static Mat4 mat4_translate(float x, float y, float z){ Mat4 r=mat4_identity(); r.m[12]=x; r.m[13]=y; r.m[14]=z; return r; }
static Mat4 mat4_rotate_y(float a){ float c=SDL_cosf(a),s=SDL_sinf(a); Mat4 r=mat4_identity(); r.m[0]=c;r.m[2]=-s;r.m[8]=s;r.m[10]=c; return r; }
static Mat4 mat4_rotate_x(float a){ float c=SDL_cosf(a),s=SDL_sinf(a); Mat4 r=mat4_identity(); r.m[5]=c;r.m[6]=s;r.m[9]=-s;r.m[10]=c; return r; }
static Mat4 mat4_perspective(float fovy,float aspect,float n,float f){ float t=1/SDL_tanf(fovy*0.5f); Mat4 r{}; r.m[0]=t/aspect; r.m[5]=t; r.m[10]=f/(n-f); r.m[11]=-1; r.m[14]=(n*f)/(n-f); return r; }
static Mat4 mat4_mul(const Mat4 &a, const Mat4 &b){ Mat4 r{}; for(int c=0;c<4;++c) for(int row=0;row<4;++row){ float s=0; for(int k=0;k<4;++k) s+=a.m[k*4+row]*b.m[c*4+k]; r.m[c*4+row]=s; } return r; }

static const char *kVS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSIn  { float3 pos [[attribute(0)]]; float4 color [[attribute(1)]]; };
struct VSOut { float4 position [[position]]; float4 color; };
struct UBO   { float4x4 mvp; };
vertex VSOut vertexMain(VSIn in [[stage_in]], constant UBO &u [[buffer(0)]]) {
    VSOut o; o.position = u.mvp * float4(in.pos, 1.0); o.color = in.color; return o;
}
)msl";
static const char *kFS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float4 color; };
fragment float4 fragmentMain(VSOut in [[stage_in]]) { return in.color; }
)msl";

struct AppState {
    SDL_Window *window=nullptr; SDL_GPUDevice *device=nullptr;
    SDL_GPUGraphicsPipeline *pipeline=nullptr;
    SDL_GPUBuffer *vbuf=nullptr, *ibuf=nullptr;
    SDL_GPUTexture *depth=nullptr;
    Uint32 depth_w=0, depth_h=0;
    Uint64 start_ns=0;
};

static SDL_AppResult fail(const char *w){ SDL_Log("%s: %s", w, SDL_GetError()); return SDL_APP_FAILURE; }
static SDL_GPUShader *make_shader(SDL_GPUDevice *d, const char *src, SDL_GPUShaderStage s, const char *e, Uint32 ubo){
    SDL_GPUShaderCreateInfo i{}; i.code=(const Uint8*)src; i.code_size=std::strlen(src);
    i.entrypoint=e; i.format=SDL_GPU_SHADERFORMAT_MSL; i.stage=s; i.num_uniform_buffers=ubo;
    return SDL_CreateGPUShader(d, &i);
}
static bool upload(SDL_GPUDevice *dev, SDL_GPUBuffer *dst, const void *src, Uint32 size){
    SDL_GPUTransferBufferCreateInfo tbi{}; tbi.usage=SDL_GPU_TRANSFERBUFFERUSAGE_UPLOAD; tbi.size=size;
    SDL_GPUTransferBuffer *tb=SDL_CreateGPUTransferBuffer(dev,&tbi);
    void *m=SDL_MapGPUTransferBuffer(dev,tb,false); std::memcpy(m,src,size); SDL_UnmapGPUTransferBuffer(dev,tb);
    SDL_GPUCommandBuffer *cmd=SDL_AcquireGPUCommandBuffer(dev);
    SDL_GPUCopyPass *cp=SDL_BeginGPUCopyPass(cmd);
    SDL_GPUTransferBufferLocation s{}; s.transfer_buffer=tb;
    SDL_GPUBufferRegion r{}; r.buffer=dst; r.size=size;
    SDL_UploadToGPUBuffer(cp,&s,&r,false);
    SDL_EndGPUCopyPass(cp); SDL_SubmitGPUCommandBuffer(cmd);
    SDL_ReleaseGPUTransferBuffer(dev,tb); return true;
}

static SDL_GPUTexture *create_depth(SDL_GPUDevice *dev, Uint32 w, Uint32 h)
{
    SDL_GPUTextureCreateInfo ti{};
    ti.type = SDL_GPU_TEXTURETYPE_2D;
    ti.format = SDL_GPU_TEXTUREFORMAT_D32_FLOAT;
    ti.usage  = SDL_GPU_TEXTUREUSAGE_DEPTH_STENCIL_TARGET;
    ti.width  = w; ti.height = h;
    ti.layer_count_or_depth = 1; ti.num_levels = 1;
    ti.sample_count = SDL_GPU_SAMPLECOUNT_1;
    return SDL_CreateGPUTexture(dev, &ti);
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Depth Testing", "1.0", "com.example.sdl3.depth");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("init");
    AppState *st = new AppState(); *appstate = st;
    st->device = SDL_CreateGPUDevice(SDL_GPU_SHADERFORMAT_MSL, true, nullptr);
    st->window = SDL_CreateWindow("Depth Testing", 800, 600, SDL_WINDOW_RESIZABLE);
    SDL_ClaimWindowForGPUDevice(st->device, st->window);

    SDL_GPUBufferCreateInfo vbi{}; vbi.usage=SDL_GPU_BUFFERUSAGE_VERTEX; vbi.size=sizeof(kCube);
    st->vbuf = SDL_CreateGPUBuffer(st->device, &vbi);
    SDL_GPUBufferCreateInfo ibi{}; ibi.usage=SDL_GPU_BUFFERUSAGE_INDEX;  ibi.size=sizeof(kIdx);
    st->ibuf = SDL_CreateGPUBuffer(st->device, &ibi);
    upload(st->device, st->vbuf, kCube, sizeof(kCube));
    upload(st->device, st->ibuf, kIdx,  sizeof(kIdx));

    SDL_GPUShader *vs = make_shader(st->device, kVS, SDL_GPU_SHADERSTAGE_VERTEX,   "vertexMain", 1);
    SDL_GPUShader *fs = make_shader(st->device, kFS, SDL_GPU_SHADERSTAGE_FRAGMENT, "fragmentMain", 0);

    SDL_GPUVertexBufferDescription vb{}; vb.slot=0; vb.pitch=sizeof(Vertex); vb.input_rate=SDL_GPU_VERTEXINPUTRATE_VERTEX;
    SDL_GPUVertexAttribute attrs[2]{};
    attrs[0].location=0; attrs[0].buffer_slot=0; attrs[0].format=SDL_GPU_VERTEXELEMENTFORMAT_FLOAT3; attrs[0].offset=offsetof(Vertex,x);
    attrs[1].location=1; attrs[1].buffer_slot=0; attrs[1].format=SDL_GPU_VERTEXELEMENTFORMAT_FLOAT4; attrs[1].offset=offsetof(Vertex,r);
    SDL_GPUVertexInputState vis{}; vis.vertex_buffer_descriptions=&vb; vis.num_vertex_buffers=1; vis.vertex_attributes=attrs; vis.num_vertex_attributes=2;

    SDL_GPUColorTargetDescription ct{}; ct.format = SDL_GetGPUSwapchainTextureFormat(st->device, st->window);

    SDL_GPUGraphicsPipelineCreateInfo pci{};
    pci.vertex_shader=vs; pci.fragment_shader=fs;
    pci.primitive_type=SDL_GPU_PRIMITIVETYPE_TRIANGLELIST;
    pci.vertex_input_state=vis;
    pci.target_info.color_target_descriptions=&ct;
    pci.target_info.num_color_targets=1;
    pci.target_info.depth_stencil_format = SDL_GPU_TEXTUREFORMAT_D32_FLOAT;
    pci.target_info.has_depth_stencil_target = true;
    pci.depth_stencil_state.enable_depth_test  = true;
    pci.depth_stencil_state.enable_depth_write = true;
    pci.depth_stencil_state.compare_op = SDL_GPU_COMPAREOP_LESS;

    st->pipeline = SDL_CreateGPUGraphicsPipeline(st->device, &pci);
    SDL_ReleaseGPUShader(st->device, vs);
    SDL_ReleaseGPUShader(st->device, fs);
    if (!st->pipeline) return fail("pipeline");

    st->start_ns = SDL_GetTicksNS();
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    float t = (SDL_GetTicksNS() - st->start_ns) / 1.0e9f;

    SDL_GPUCommandBuffer *cmd = SDL_AcquireGPUCommandBuffer(st->device);
    SDL_GPUTexture *swap=nullptr; Uint32 sw=0, sh=0;
    SDL_WaitAndAcquireGPUSwapchainTexture(cmd, st->window, &swap, &sw, &sh);

    if (swap) {
        // (Re)create the depth texture if size changed.
        if (sw != st->depth_w || sh != st->depth_h) {
            if (st->depth) SDL_ReleaseGPUTexture(st->device, st->depth);
            st->depth   = create_depth(st->device, sw, sh);
            st->depth_w = sw; st->depth_h = sh;
        }

        float aspect = sh > 0 ? (float)sw / (float)sh : 1.0f;
        Mat4 model = mat4_mul(mat4_rotate_y(t), mat4_rotate_x(t * 0.6f));
        Mat4 view  = mat4_translate(0, 0, -3.0f);
        Mat4 proj  = mat4_perspective(60.0f * SDL_PI_F / 180.0f, aspect, 0.1f, 100.0f);
        Mat4 mvp   = mat4_mul(proj, mat4_mul(view, model));

        SDL_GPUColorTargetInfo cti{};
        cti.texture=swap; cti.clear_color=SDL_FColor{0.05f,0.07f,0.10f,1};
        cti.load_op=SDL_GPU_LOADOP_CLEAR; cti.store_op=SDL_GPU_STOREOP_STORE;

        SDL_GPUDepthStencilTargetInfo dti{};
        dti.texture = st->depth;
        dti.clear_depth = 1.0f;
        dti.load_op = SDL_GPU_LOADOP_CLEAR;
        dti.store_op = SDL_GPU_STOREOP_DONT_CARE;

        SDL_GPURenderPass *pass = SDL_BeginGPURenderPass(cmd, &cti, 1, &dti);
        SDL_BindGPUGraphicsPipeline(pass, st->pipeline);
        SDL_GPUBufferBinding vbind{}; vbind.buffer=st->vbuf; SDL_BindGPUVertexBuffers(pass,0,&vbind,1);
        SDL_GPUBufferBinding ibind{}; ibind.buffer=st->ibuf; SDL_BindGPUIndexBuffer(pass,&ibind, SDL_GPU_INDEXELEMENTSIZE_16BIT);
        SDL_PushGPUVertexUniformData(cmd, 0, &mvp, sizeof(mvp));
        SDL_DrawGPUIndexedPrimitives(pass, 36, 1, 0, 0, 0);
        SDL_EndGPURenderPass(pass);
    }
    SDL_SubmitGPUCommandBuffer(cmd);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (st->device) {
        if (st->depth) SDL_ReleaseGPUTexture(st->device, st->depth);
        if (st->ibuf)  SDL_ReleaseGPUBuffer(st->device, st->ibuf);
        if (st->vbuf)  SDL_ReleaseGPUBuffer(st->device, st->vbuf);
        SDL_ReleaseGPUGraphicsPipeline(st->device, st->pipeline);
        if (st->window) SDL_ReleaseWindowFromGPUDevice(st->device, st->window);
        SDL_DestroyGPUDevice(st->device);
    }
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Disable depth-write but keep depth-test. The cube still renders correctly until any face draws into a place that was already further — then ghost faces reappear.
  2. Change compare_op to GREATER. Now far faces hide near ones — the cube looks inside-out.
  3. Try SDL_GPU_TEXTUREFORMAT_D24_UNORM_S8_UINT (24-bit depth + 8-bit stencil). Same picture, different memory.
