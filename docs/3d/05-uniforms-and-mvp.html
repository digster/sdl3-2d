Recipe 05 — Uniforms & the MVP Matrix

Push a 4×4 matrix into the vertex shader and watch the quad spin. The model-view-projection pipeline, hand-built from scratch.

What you'll learn

  • The three matrices that move geometry from object space to screen space.
  • How to construct rotation, translation, and perspective matrices by hand (no GLM).
  • SDL_GPU's push constants: SDL_PushGPUVertexUniformData, the cheapest per-frame uniform.
  • How the same shader receives the matrix on the MSL side.

Prerequisites

Recipe 04 — Index Buffers. Some linear algebra recall helps but isn't required.

Walkthrough

What "MVP" means

A vertex starts in the object's local coordinate system. Three transformations carry it to screen space:

  • Model — where the object sits in the world (translation, rotation, scale).
  • View — where the camera is and what it's looking at (inverse of the camera's transform).
  • Projection — how the camera maps 3D to 2D (perspective foreshortening).

Multiplying them in order — MVP = P * V * M — lets the shader do one matrix–vector multiply per vertex.

A tiny mat4 in pure C++

For pedagogy we'll skip GLM and write a 16-float column-major matrix with three constructors. Metal (and Vulkan, and D3D12) are column-major; matching that convention avoids transposes later. 

struct Mat4 {
    float m[16];   // column-major: m[col*4 + row]
};

static Mat4 mat4_identity() {
    Mat4 r{}; r.m[0] = r.m[5] = r.m[10] = r.m[15] = 1.0f; return r;
}

static Mat4 mat4_translate(float x, float y, float z) {
    Mat4 r = mat4_identity();
    r.m[12] = x; r.m[13] = y; r.m[14] = z;
    return r;
}

static Mat4 mat4_rotate_y(float radians) {
    float c = SDL_cosf(radians), s = SDL_sinf(radians);
    Mat4 r = mat4_identity();
    r.m[0]  =  c; r.m[2]  = -s;
    r.m[8]  =  s; r.m[10] =  c;
    return r;
}

// Right-handed perspective. fov_y in radians.
static Mat4 mat4_perspective(float fov_y, float aspect, float n, float f) {
    float t = 1.0f / SDL_tanf(fov_y * 0.5f);
    Mat4 r{};
    r.m[0]  = t / aspect;
    r.m[5]  = t;
    r.m[10] = f / (n - f);
    r.m[11] = -1.0f;
    r.m[14] = (n * f) / (n - f);
    return r;
}

static Mat4 mat4_mul(const Mat4 &a, const Mat4 &b) {
    Mat4 r{};
    for (int col = 0; col < 4; ++col)
        for (int row = 0; row < 4; ++row) {
            float s = 0.0f;
            for (int k = 0; k < 4; ++k)
                s += a.m[k * 4 + row] * b.m[col * 4 + k];
            r.m[col * 4 + row] = s;
        }
    return r;
}

Building MVP each frame

// Inside SDL_AppIterate, after computing `t` (seconds since start):
int win_w = 0, win_h = 0; SDL_GetWindowSize(st->window, &win_w, &win_h);
float aspect = (float)win_w / (float)win_h;

Mat4 model      = mat4_rotate_y(t);                    // spin the quad
Mat4 view       = mat4_translate(0, 0, -2.0f);         // camera 2 units back
Mat4 projection = mat4_perspective(60.0f * SDL_PI_F / 180.0f, aspect, 0.1f, 100.0f);

Mat4 mvp = mat4_mul(projection, mat4_mul(view, model));

Pushing it to the shader

SDL_GPU's push constants are the lightest uniform mechanism — no buffer creation, the data goes inline with the command buffer: 

SDL_PushGPUVertexUniformData(cmd, /*slot=*/0, &mvp, sizeof(mvp));

The shader receives it via [[buffer(0)]] in the buffer-binding section of the vertex stage (slot 0 by convention here).

Receiving in MSL

struct UBO { float4x4 mvp; };

vertex VSOut vertexMain(VSIn in [[stage_in]],
                        constant UBO &u [[buffer(0)]])
{
    VSOut o;
    o.position = u.mvp * float4(in.pos, 1.0);
    o.color    = in.color;
    return o;
}
Why column-major. Metal, Vulkan, and D3D12 all expect column-major matrices in shaders. Authoring them column-major in C++ means you can memcpy straight in with no transpose. (If you use GLM later, that's its default too.)

Complete listing

/*
 * 05-uniforms-and-mvp.cpp — the quad from recipe 04 now spinning in 3D.
 */
#define SDL_MAIN_USE_CALLBACKS 1
#include <SDL3/SDL.h>
#include <SDL3/SDL_main.h>

#include <cstring>
#include <cstddef>

struct Vertex { float x, y, z, r, g, b, a; };
static const Vertex kQuad[4] = {
    { -0.5f,  0.5f, 0, 1, 0, 0, 1 },
    {  0.5f,  0.5f, 0, 0, 1, 0, 1 },
    {  0.5f, -0.5f, 0, 0, 0, 1, 1 },
    { -0.5f, -0.5f, 0, 1, 1, 0, 1 },
};
static const Uint16 kIndices[6] = { 0, 1, 2,  0, 2, 3 };

struct Mat4 { float m[16]; };
static Mat4 mat4_identity() { Mat4 r{}; r.m[0]=r.m[5]=r.m[10]=r.m[15]=1; return r; }
static Mat4 mat4_translate(float x, float y, float z) {
    Mat4 r = mat4_identity(); r.m[12]=x; r.m[13]=y; r.m[14]=z; return r;
}
static Mat4 mat4_rotate_y(float a) {
    float c = SDL_cosf(a), s = SDL_sinf(a);
    Mat4 r = mat4_identity(); r.m[0]=c; r.m[2]=-s; r.m[8]=s; r.m[10]=c; return r;
}
static Mat4 mat4_perspective(float fovy, float aspect, float n, float f) {
    float t = 1.0f / SDL_tanf(fovy * 0.5f);
    Mat4 r{}; r.m[0] = t / aspect; r.m[5] = t;
    r.m[10] = f / (n - f); r.m[11] = -1; r.m[14] = (n * f) / (n - f);
    return r;
}
static Mat4 mat4_mul(const Mat4 &a, const Mat4 &b) {
    Mat4 r{};
    for (int c=0;c<4;++c) for (int row=0;row<4;++row) {
        float s=0; for(int k=0;k<4;++k) s += a.m[k*4+row]*b.m[c*4+k];
        r.m[c*4+row] = s;
    }
    return r;
}

static const char *kVS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSIn  { float3 pos [[attribute(0)]]; float4 color [[attribute(1)]]; };
struct VSOut { float4 position [[position]]; float4 color; };
struct UBO   { float4x4 mvp; };
vertex VSOut vertexMain(VSIn in [[stage_in]], constant UBO &u [[buffer(0)]]) {
    VSOut o; o.position = u.mvp * float4(in.pos, 1.0); o.color = in.color; return o;
}
)msl";
static const char *kFS = R"msl(
#include <metal_stdlib>
using namespace metal;
struct VSOut { float4 position [[position]]; float4 color; };
fragment float4 fragmentMain(VSOut in [[stage_in]]) { return in.color; }
)msl";

struct AppState {
    SDL_Window *window=nullptr; SDL_GPUDevice *device=nullptr;
    SDL_GPUGraphicsPipeline *pipeline=nullptr;
    SDL_GPUBuffer *vbuf=nullptr, *ibuf=nullptr;
    Uint64 start_ns = 0;
};

static SDL_AppResult fail(const char *w){ SDL_Log("%s: %s", w, SDL_GetError()); return SDL_APP_FAILURE; }
static SDL_GPUShader *make_shader(SDL_GPUDevice *d, const char *src, SDL_GPUShaderStage s, const char *e,
                                  Uint32 uniforms = 0)
{
    SDL_GPUShaderCreateInfo i{};
    i.code = (const Uint8 *)src; i.code_size = std::strlen(src);
    i.entrypoint = e; i.format = SDL_GPU_SHADERFORMAT_MSL; i.stage = s;
    i.num_uniform_buffers = uniforms;
    return SDL_CreateGPUShader(d, &i);
}
static bool upload(SDL_GPUDevice *dev, SDL_GPUBuffer *dst, const void *src, Uint32 size) {
    SDL_GPUTransferBufferCreateInfo tbi{}; tbi.usage = SDL_GPU_TRANSFERBUFFERUSAGE_UPLOAD; tbi.size = size;
    SDL_GPUTransferBuffer *tb = SDL_CreateGPUTransferBuffer(dev, &tbi);
    void *m = SDL_MapGPUTransferBuffer(dev, tb, false);
    std::memcpy(m, src, size);
    SDL_UnmapGPUTransferBuffer(dev, tb);
    SDL_GPUCommandBuffer *cmd = SDL_AcquireGPUCommandBuffer(dev);
    SDL_GPUCopyPass *cp = SDL_BeginGPUCopyPass(cmd);
    SDL_GPUTransferBufferLocation s{}; s.transfer_buffer = tb;
    SDL_GPUBufferRegion r{}; r.buffer = dst; r.size = size;
    SDL_UploadToGPUBuffer(cp, &s, &r, false);
    SDL_EndGPUCopyPass(cp); SDL_SubmitGPUCommandBuffer(cmd);
    SDL_ReleaseGPUTransferBuffer(dev, tb);
    return true;
}

SDL_AppResult SDL_AppInit(void **appstate, int, char *[])
{
    SDL_SetAppMetadata("Uniforms & MVP", "1.0", "com.example.sdl3.mvp");
    if (!SDL_Init(SDL_INIT_VIDEO)) return fail("init");
    AppState *st = new AppState(); *appstate = st;
    st->device = SDL_CreateGPUDevice(SDL_GPU_SHADERFORMAT_MSL, true, nullptr);
    st->window = SDL_CreateWindow("Uniforms & MVP", 800, 600, SDL_WINDOW_RESIZABLE);
    SDL_ClaimWindowForGPUDevice(st->device, st->window);

    SDL_GPUBufferCreateInfo vbi{}; vbi.usage = SDL_GPU_BUFFERUSAGE_VERTEX; vbi.size = sizeof(kQuad);
    st->vbuf = SDL_CreateGPUBuffer(st->device, &vbi);
    SDL_GPUBufferCreateInfo ibi{}; ibi.usage = SDL_GPU_BUFFERUSAGE_INDEX;  ibi.size = sizeof(kIndices);
    st->ibuf = SDL_CreateGPUBuffer(st->device, &ibi);
    upload(st->device, st->vbuf, kQuad,    sizeof(kQuad));
    upload(st->device, st->ibuf, kIndices, sizeof(kIndices));

    // Note: vertex shader uses 1 uniform buffer slot (the MVP).
    SDL_GPUShader *vs = make_shader(st->device, kVS, SDL_GPU_SHADERSTAGE_VERTEX,   "vertexMain", 1);
    SDL_GPUShader *fs = make_shader(st->device, kFS, SDL_GPU_SHADERSTAGE_FRAGMENT, "fragmentMain", 0);

    SDL_GPUVertexBufferDescription vb{}; vb.slot=0; vb.pitch=sizeof(Vertex); vb.input_rate=SDL_GPU_VERTEXINPUTRATE_VERTEX;
    SDL_GPUVertexAttribute attrs[2]{};
    attrs[0].location=0; attrs[0].buffer_slot=0; attrs[0].format=SDL_GPU_VERTEXELEMENTFORMAT_FLOAT3; attrs[0].offset=offsetof(Vertex,x);
    attrs[1].location=1; attrs[1].buffer_slot=0; attrs[1].format=SDL_GPU_VERTEXELEMENTFORMAT_FLOAT4; attrs[1].offset=offsetof(Vertex,r);
    SDL_GPUVertexInputState vis{};
    vis.vertex_buffer_descriptions = &vb; vis.num_vertex_buffers = 1;
    vis.vertex_attributes = attrs;        vis.num_vertex_attributes = 2;

    SDL_GPUColorTargetDescription ct{};
    ct.format = SDL_GetGPUSwapchainTextureFormat(st->device, st->window);

    SDL_GPUGraphicsPipelineCreateInfo pci{};
    pci.vertex_shader = vs; pci.fragment_shader = fs;
    pci.primitive_type = SDL_GPU_PRIMITIVETYPE_TRIANGLELIST;
    pci.vertex_input_state = vis;
    pci.target_info.color_target_descriptions = &ct;
    pci.target_info.num_color_targets = 1;
    st->pipeline = SDL_CreateGPUGraphicsPipeline(st->device, &pci);
    SDL_ReleaseGPUShader(st->device, vs);
    SDL_ReleaseGPUShader(st->device, fs);
    if (!st->pipeline) return fail("pipeline");

    st->start_ns = SDL_GetTicksNS();
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppEvent(void *, SDL_Event *e)
{
    if (e->type == SDL_EVENT_QUIT) return SDL_APP_SUCCESS;
    if (e->type == SDL_EVENT_KEY_DOWN && e->key.scancode == SDL_SCANCODE_ESCAPE) return SDL_APP_SUCCESS;
    return SDL_APP_CONTINUE;
}

SDL_AppResult SDL_AppIterate(void *appstate)
{
    AppState *st = static_cast<AppState *>(appstate);
    float t = (SDL_GetTicksNS() - st->start_ns) / 1.0e9f;

    int win_w = 0, win_h = 0; SDL_GetWindowSize(st->window, &win_w, &win_h);
    float aspect = win_h > 0 ? (float)win_w / (float)win_h : 1.0f;

    Mat4 model = mat4_rotate_y(t);
    Mat4 view  = mat4_translate(0, 0, -2.0f);
    Mat4 proj  = mat4_perspective(60.0f * SDL_PI_F / 180.0f, aspect, 0.1f, 100.0f);
    Mat4 mvp   = mat4_mul(proj, mat4_mul(view, model));

    SDL_GPUCommandBuffer *cmd = SDL_AcquireGPUCommandBuffer(st->device);
    SDL_GPUTexture *swap = nullptr; Uint32 sw=0, sh=0;
    SDL_WaitAndAcquireGPUSwapchainTexture(cmd, st->window, &swap, &sw, &sh);

    if (swap) {
        SDL_GPUColorTargetInfo cti{};
        cti.texture = swap;
        cti.clear_color = SDL_FColor{ 0.05f, 0.07f, 0.10f, 1 };
        cti.load_op = SDL_GPU_LOADOP_CLEAR; cti.store_op = SDL_GPU_STOREOP_STORE;
        SDL_GPURenderPass *pass = SDL_BeginGPURenderPass(cmd, &cti, 1, nullptr);
        SDL_BindGPUGraphicsPipeline(pass, st->pipeline);
        SDL_GPUBufferBinding vbind{}; vbind.buffer = st->vbuf;
        SDL_BindGPUVertexBuffers(pass, 0, &vbind, 1);
        SDL_GPUBufferBinding ibind{}; ibind.buffer = st->ibuf;
        SDL_BindGPUIndexBuffer(pass, &ibind, SDL_GPU_INDEXELEMENTSIZE_16BIT);
        SDL_PushGPUVertexUniformData(cmd, 0, &mvp, sizeof(mvp));
        SDL_DrawGPUIndexedPrimitives(pass, 6, 1, 0, 0, 0);
        SDL_EndGPURenderPass(pass);
    }
    SDL_SubmitGPUCommandBuffer(cmd);
    return SDL_APP_CONTINUE;
}

void SDL_AppQuit(void *appstate, SDL_AppResult)
{
    AppState *st = static_cast<AppState *>(appstate);
    if (!st) return;
    if (st->device) {
        if (st->ibuf) SDL_ReleaseGPUBuffer(st->device, st->ibuf);
        if (st->vbuf) SDL_ReleaseGPUBuffer(st->device, st->vbuf);
        SDL_ReleaseGPUGraphicsPipeline(st->device, st->pipeline);
        if (st->window) SDL_ReleaseWindowFromGPUDevice(st->device, st->window);
        SDL_DestroyGPUDevice(st->device);
    }
    SDL_DestroyWindow(st->window);
    delete st;
}

Try it yourself

  1. Move the camera: change view to mat4_translate(0, 0, -4.0f). The quad shrinks.
  2. Rotate around X or Z instead of Y: write mat4_rotate_x by analogy.
  3. Stop rotating: replace model with mat4_identity() — the quad now sits flat-on, which means you only see its 2D outline (no thickness). This is the moment to switch to the cube in recipe 06.
  4. Swap the multiply order to P * M * V — the picture goes wrong in revealing ways.
